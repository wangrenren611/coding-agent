## 1. 模块概述

### 1.1 设计目标

会话管理模块是企业级 Agent 服务端的核心组件，负责管理用户与 Agent 之间的对话上下文。本模块需要实现以下目标：

| 目标 | 描述 | 优先级 |
|------|------|--------|
| **一致性** | 保证并发场景下的数据一致性，避免消息丢失或重复 | P0 |
| **高性能** | 低延迟的上下文读写，支持高并发 | P0 |
| **可扩展** | 支持水平扩展，适应流量增长 | P0 |
| **持久化** | 确保消息不因服务重启而丢失 | P0 |
| **压缩** | 自动管理上下文长度，控制 Token 消耗 | P1 |

### 1.2 核心职责

```
会话管理模块
│
├── Session Service (会话服务)
│   ├── 创建/恢复/归档/删除会话
│   ├── 会话配置管理
│   └── 会话状态追踪
│
├── Context Manager (上下文管理)
│   ├── 消息存储与检索
│   ├── 乐观锁版本控制
│   └── 上下文压缩
│
├── Message Queue (消息队列)
│   ├── 消息操作串行化
│   ├── 异步持久化
│   └── 并发控制
│
└── Compaction Service (压缩服务)
    ├── Token 计算
    ├── 压缩策略
    └── 摘要生成
```

### 3.2 实现设计

```typescript
/**
 * Session Service 实现
 * 
 * 设计要点：
 * 1. 无状态设计 - 所有状态存储在 Redis/PostgreSQL
 * 2. 缓存热点会话 - 使用 Redis 缓存活跃会话
 * 3. 延迟删除 - 标记删除而非立即物理删除
 */
class SessionService implements ISessionService {
  private redis: Redis;
  private db: PrismaClient;
  private cache: SessionCache;
  
  // 缓存配置
  private readonly CACHE_TTL = 3600;        // 1小时
  private readonly CACHE_MAX_SIZE = 10000;  // 最大缓存数
  
  /**
   * 创建会话
   */
  async create(params: CreateSessionParams): Promise<Session> {
    // 1. 生成会话ID
    const sessionId = generateUUID();
    
    // 2. 验证租户配额
    await this.checkTenantQuota(params.tenantId);
    
    // 3. 创建会话记录
    const session: Session = {
      id: sessionId,
      tenantId: params.tenantId,
      userId: params.userId,
      config: {
        systemPrompt: params.systemPrompt,
        model: params.model || DEFAULT_MODEL,
        tools: params.tools || [],
        temperature: params.temperature,
        maxTokens: params.maxTokens,
        metadata: params.metadata || {},
      },
      status: 'active',
      usage: {
        promptTokens: 0,
        completionTokens: 0,
        totalTokens: 0,
        requestCount: 0,
        messageCount: 0,
      },
      createdAt: new Date(),
      updatedAt: new Date(),
      lastActiveAt: new Date(),
      expiresAt: params.expiresIn 
        ? new Date(Date.now() + params.expiresIn * 1000) 
        : undefined,
    };
    
    // 4. 持久化到数据库
    await this.db.session.create({
      data: this.toDbSession(session),
    });
    
    // 5. 初始化空上下文
    await this.contextService.initializeContext(sessionId, params.systemPrompt);
    
    // 6. 缓存会话
    await this.cache.set(sessionId, session, this.CACHE_TTL);
    
    // 7. 记录指标
    this.metrics.increment('session.created', { 
      tenantId: params.tenantId 
    });
    
    return session;
  }
  
  /**
   * 获取会话
   * 
   * 策略：
   * 1. 先查缓存
   * 2. 缓存未命中则查数据库
   * 3. 存入缓存
   */
  async get(sessionId: string): Promise<Session | null> {
    // 1. 尝试缓存
    const cached = await this.cache.get(sessionId);
    if (cached) {
      return cached;
    }
    
    // 2. 查询数据库
    const dbSession = await this.db.session.findUnique({
      where: { id: sessionId },
    });
    
    if (!dbSession || dbSession.status === 'deleted') {
      return null;
    }
    
    // 3. 转换并缓存
    const session = this.fromDbSession(dbSession);
    await this.cache.set(sessionId, session, this.CACHE_TTL);
    
    return session;
  }
  
  /**
   * 恢复会话
   * 
   * 恢复会话时需要：
   * 1. 验证会话存在
   * 2. 检查会话未过期
   * 3. 更新最后活跃时间
   */
  async resume(sessionId: string): Promise<Session> {
    const session = await this.get(sessionId);
    
    if (!session) {
      throw new NotFoundError(`Session ${sessionId} not found`);
    }
    
    if (session.status === 'deleted') {
      throw new InvalidOperationError('Cannot resume deleted session');
    }
    
    if (session.status === 'archived') {
      // 从归档恢复
      await this.update(sessionId, { status: 'active' });
    }
    
    // 更新最后活跃时间
    await this.updateLastActive(sessionId);
    
    return session;
  }
  
  /**
   * 删除会话 (软删除)
   */
  async delete(sessionId: string): Promise<void> {
    const session = await this.get(sessionId);
    if (!session) {
      throw new NotFoundError(`Session ${sessionId} not found`);
    }
    
    // 软删除
    await this.update(sessionId, { status: 'deleted' });
    
    // 清除缓存
    await this.cache.delete(sessionId);
    
    // 记录指标
    this.metrics.increment('session.deleted', { 
      tenantId: session.tenantId 
    });
  }
  
  /**
   * 计量 Token 使用
   */
  async recordUsage(sessionId: string, usage: Usage): Promise<void> {
    // 更新数据库
    await this.db.session.update({
      where: { id: sessionId },
      data: {
        usage: {
          promptTokens: { increment: usage.promptTokens },
          completionTokens: { increment: usage.completionTokens || 0 },
          totalTokens: { increment: usage.totalTokens },
          requestCount: { increment: 1 },
        },
        updatedAt: new Date(),
      },
    });
    
    // 更新缓存 (如果存在)
    const cached = await this.cache.get(sessionId);
    if (cached) {
      cached.usage.promptTokens += usage.promptTokens;
      cached.usage.completionTokens += usage.completionTokens || 0;
      cached.usage.totalTokens += usage.totalTokens;
      cached.usage.requestCount += 1;
      await this.cache.set(sessionId, cached, this.CACHE_TTL);
    }
    
    // 记录到使用统计
    await this.usageService.record(sessionId, usage);
  }
  
  /**
   * 检查租户配额
   */
  private async checkTenantQuota(tenantId: string): Promise<void> {
    const quota = await this.quotaService.getTenantQuota(tenantId);
    const current = await this.sessionRepository.countActiveSessions(tenantId);
    
    if (current >= quota.maxSessions) {
      throw new QuotaExceededError('Maximum sessions limit reached');
    }
  }
}
```

### 4.2 缓存策略

```typescript
/**
 * Context Manager 实现
 * 
 * 设计要点：
 * 1. Redis 缓存当前上下文 (热点数据)
 * 2. PostgreSQL 持久化完整数据
 * 3. 使用版本号实现乐观锁
 * 4. 定期刷新缓存到数据库
 */
class ContextService implements IContextService {
  private redis: Redis;
  private db: PrismaClient;
  
  // 缓存键前缀
  private readonly CACHE_PREFIX = 'context:';
  private readonly VERSION_PREFIX = 'context:version:';
  
  // 缓存配置
  private readonly CACHE_TTL = 300;        // 5分钟
  private readonly MAX_CONTEXT_TOKENS = 200000;
  
  /**
   * 获取上下文
   * 
   * 策略：
   * 1. 优先从 Redis 读取
   * 2. 未命中则从 PostgreSQL 读取并缓存
   * 3. 验证版本号防止脏读
   */
  async getContext(sessionId: string): Promise<Context> {
    const cacheKey = `${this.CACHE_PREFIX}${sessionId}`;
    
    // 1. 尝试 Redis
    const cached = await this.redis.get(cacheKey);
    if (cached) {
      return JSON.parse(cached);
    }
    
    // 2. 查询数据库
    const dbContext = await this.db.context.findUnique({
      where: { sessionId },
      include: { messages: true },
    });
    
    if (!dbContext) {
      throw new NotFoundError(`Context for session ${sessionId} not found`);
    }
    
    // 3. 转换并缓存
    const context = this.fromDbContext(dbContext);
    await this.redis.setex(cacheKey, this.CACHE_TTL, JSON.stringify(context));
    
    return context;
  }
  
  /**
   * 乐观锁更新上下文
   * 
   * 实现：
   * 1. 读取当前版本号
   * 2. 执行更新，WHERE 版本号 = 当前版本号
   * 3. 如果更新行数为 0，说明有并发冲突，回退并重试
   */
  async updateContext(
    sessionId: string, 
    updates: ContextUpdates,
    expectedVersion?: number
  ): Promise<Context> {
    const versionKey = `${this.VERSION_PREFIX}${sessionId}`;
    
    // 1. 获取当前版本 (或使用传入的版本)
    const currentVersion = expectedVersion 
      ?? await this.redis.get(versionKey)
      ?? await this.getCurrentVersionFromDb(sessionId);
    
    const newVersion = currentVersion + 1;
    
    // 2. 构建更新数据
    const updateData: any = {
      version: newVersion,
      updatedAt: new Date(),
    };
    
    if (updates.messages) {
      updateData.messages = {
        deleteMany: { sessionId },
        create: updates.messages.map(m => this.toDbMessage(m)),
      };
      updateData.stats = {
        totalTokens: this.estimateTokens(updates.messages),
        messageCount: updates.messages.length,
      };
    }
    
    if (updates.stats) {
      updateData.stats = updates.stats;
    }
    
    // 3. 执行更新 (乐观锁)
    try {
      await this.db.$transaction(async (tx) => {
        // 先检查版本
        const current = await tx.context.findUnique({
          where: { sessionId },
          select: { version: true },
        });
        
        if (current?.version !== currentVersion) {
          throw new ConflictError('Context version mismatch');
        }
        
        // 更新
        await tx.context.update({
          where: { sessionId },
          data: updateData,
        });
      });
    } catch (error) {
      if (error instanceof ConflictError) {
        // 冲突，回退并重试
        await this.cache.invalidate(sessionId);
        return this.updateContext(sessionId, updates);
      }
      throw error;
    }
    
    // 4. 更新 Redis 缓存
    await this.redis.set(versionKey, newVersion.toString());
    await this.invalidateCache(sessionId);
    
    // 5. 返回新上下文
    return this.getContext(sessionId);
  }
  
  /**
   * 获取当前版本号 (从数据库)
   */
  private async getCurrentVersionFromDb(sessionId: string): Promise<number> {
    const ctx = await this.db.context.findUnique({
      where: { sessionId },
      select: { version: true },
    });
    return ctx?.version ?? 0;
  }
  
  /**
   * 清除缓存
   */
  private async invalidateCache(sessionId: string): Promise<void> {
    const cacheKey = `${this.CACHE_PREFIX}${sessionId}`;
    await this.redis.del(cacheKey);
  }
}
```

## 5. Message Queue (消息队列)

### 5.1 设计原理

消息队列是保证**单会话消息操作串行化**的核心组件。

```
问题场景：
- 用户快速发送两条消息 A 和 B
- 如果没有队列，A 和 B 可能同时写入
- 由于消息的 timestamp 或 ID 可能相同或交叉
- 导致消息顺序不确定或数据竞争

解决方案：
- 每个会话一个独立的 Redis List 队列
- 所有消息操作必须入队
- 消费者串行处理队列
```

### 5.2 核心接口

```typescript
// 消息队列服务接口
interface IMessageQueueService {
  // 添加消息 (入队)
  enqueue(sessionId: string, message: Message): Promise<void>;
  
  // 获取消息 (不出队)
  peek(sessionId: string, count?: number): Promise<Message[]>;
  
  // 出队 (消费者)
  dequeue(sessionId: string): Promise<Message | null>;
  
  // 获取队列长度
  size(sessionId: string): Promise<number>;
  
  // 清空队列
  clear(sessionId: string): Promise<void>;
}

// 队列消息包装
interface QueuedMessage {
  message: Message;
  operation: 'add' | 'update' | 'remove';
  timestamp: number;
  retryCount: number;
}
```

### 5.3 实现设计

```typescript
/**
 * Message Queue Service 实现
 * 
 * 设计要点：
 * 1. 使用 Redis List 作为队列
 * 2. 每个会话独立的队列键
 * 3. 消费者模式异步持久化
 * 4. 支持重试机制
 */
class MessageQueueService implements IMessageQueueService {
  private redis: Redis;
  private db: PrismaClient;
  
  // 队列键前缀
  private readonly QUEUE_PREFIX = 'queue:session:';
  
  // 配置
  private readonly MAX_RETRY = 3;
  private readonly RETRY_DELAY = 1000;  // ms
  private readonly BATCH_SIZE = 10;
  
  // 消费者实例
  private consumer: QueueConsumer;
  
  /**
   * 添加消息到队列
   * 
   * 使用 RPUSH 将消息添加到队列尾部
   * 异步触发消费者处理
   */
  async enqueue(sessionId: string, message: Message): Promise<void> {
    const queueKey = this.getQueueKey(sessionId);
    
    const queuedMessage: QueuedMessage = {
      message,
      operation: 'add',
      timestamp: Date.now(),
      retryCount: 0,
    };
    
    // 1. 加入队列 (原子操作)
    await this.redis.rpush(queueKey, JSON.stringify(queuedMessage));
    
    // 2. 设置过期时间 (24小时)
    await this.redis.expire(queueKey, 86400);
    
    // 3. 通知消费者处理
    await this.consumer.notify(sessionId);
    
    // 4. 更新会话最后活跃时间
    await this.sessionService.updateLastActive(sessionId);
  }
  
  /**
   * 获取队列长度
   */
  async size(sessionId: string): Promise<number> {
    const queueKey = this.getQueueKey(sessionId);
    return this.redis.llen(queueKey);
  }
  
  /**
   * 获取队列前端消息 (不出队)
   */
  async peek(sessionId: string, count: number = 10): Promise<Message[]> {
    const queueKey = this.getQueueKey(sessionId);
    const items = await this.redis.lrange(queueKey, 0, count - 1);
    
    return items.map(item => {
      const queued = JSON.parse(item) as QueuedMessage;
      return queued.message;
    });
  }
  
  /**
   * 清空队列
   */
  async clear(sessionId: string): Promise<void> {
    const queueKey = this.getQueueKey(sessionId);
    await this.redis.del(queueKey);
  }
  
  /**
   * 获取队列键
   */
  private getQueueKey(sessionId: string): string {
    return `${this.QUEUE_PREFIX}${sessionId}`;
  }
}

/**
 * 队列消费者
 * 
 * 负责从队列中取出消息并持久化到数据库
 */
class QueueConsumer {
  private redis: Redis;
  private db: PrismaClient;
  private processing = new Set<string>();  // 正在处理的会话
  
  /**
   * 处理指定会话的队列
   */
  async processQueue(sessionId: string): Promise<void> {
    // 防止同一会话并发处理
    if (this.processing.has(sessionId)) {
      return;
    }
    
    this.processing.add(sessionId);
    
    try {
      const queueKey = `queue:session:${sessionId}`;
      
      while (true) {
        // 1. 从队列头部取出一条消息
        const item = await this.redis.lpop(queueKey);
        if (!item) {
          break;  // 队列为空
        }
        
        const queued = JSON.parse(item) as QueuedMessage;
        
        try {
          // 2. 持久化到数据库
          await this.persistMessage(sessionId, queued);
        } catch (error) {
          // 3. 处理失败，加入重试队列
          await this.handleFailure(sessionId, queued, error);
        }
      }
    } finally {
      this.processing.delete(sessionId);
    }
  }
  
  /**
   * 持久化消息
   */
  private async persistMessage(
    sessionId: string, 
    queued: QueuedMessage
  ): Promise<void> {
    const { message, operation } = queued;
    
    await this.db.$transaction(async (tx) => {
      switch (operation) {
        case 'add':
          // 添加消息到上下文
          await tx.context.update({
            where: { sessionId },
            data: {
              messages: {
                create: this.toDbMessage(message),
              },
              version: { increment: 1 },
              updatedAt: new Date(),
            },
          });
          break;
          
        case 'update':
          await tx.message.update({
            where: { id: message.id },
            data: this.toDbMessage(message),
          });
          break;
          
        case 'remove':
          await tx.message.update({
            where: { id: message.id },
            data: { 
              excludedFromContext: true,
              excludedReason: 'manual',
            },
          });
          break;
      }
    });
  }
  
  /**
   * 处理失败 - 重试逻辑
   */
  private async handleFailure(
    sessionId: string,
    queued: QueuedMessage,
    error: Error
  ): Promise<void> {
    if (queued.retryCount < MAX_RETRY) {
      // 重新加入队列，延迟处理
      queued.retryCount++;
      const queueKey = `queue:session:${sessionId}`;
      
      // 使用 Redis 延迟队列 (Sorted Set)
      const delayedKey = `queue:delayed:${sessionId}`;
      const delay = RETRY_DELAY * Math.pow(2, queued.retryCount - 1);
      
      await this.redis.zadd(
        delayedKey, 
        Date.now() + delay,
        JSON.stringify(queued)
      );
    } else {
      // 重试次数用完，记录错误日志
      this.logger.error('Message queue retry exhausted', {
        sessionId,
        messageId: queued.message.id,
        error: error.message,
      });
      
      // 发送告警
      await this.alertService.send({
        type: 'queue_failure',
        sessionId,
        messageId: queued.message.id,
        error: error.message,
      });
    }
  }
}
```

## 6. Compaction Service (压缩服务)

### 6.1 设计原理

```
上下文压缩流程：

1. 触发条件
   ├── Token 使用超过阈值 (如 90%)
   ├── 手动触发
   └── 定时任务

2. 压缩步骤
   ├── 分离消息区域
   │   ├── system (保留)
   │   ├── pending (待压缩)
   │   └── active (保留区)
   │
   ├── 处理工具调用配对
   │   └── 确保 pending 区不包含未完成的 tool_calls
   │
   ├── 生成摘要
   │   └── 调用 LLM 生成摘要
   │
   └── 重组消息
       └── system + summary + active

3. 持久化
   ├── 保存摘要消息
   ├── 标记被压缩消息
   └── 创建压缩记录
```

### 6.2 核心接口

```typescript
// 压缩服务接口
interface ICompactionService {
  // 检查是否需要压缩
  shouldCompact(sessionId: string): Promise<boolean>;
  
  // 执行压缩
  compact(sessionId: string): Promise<CompactionResult>;
  
  // 获取 Token 信息
  getTokenInfo(sessionId: string): Promise<TokenInfo>;
  
  // 手动触发压缩
  forceCompact(sessionId: string): Promise<CompactionResult>;
}

// Token 信息
interface TokenInfo {
  estimatedTotal: number;
  usableLimit: number;
  threshold: number;
  shouldCompact: boolean;
  messageCount: number;
}
```

### 6.3 实现设计

```typescript
/**
 * Compaction Service 实现
 */
class CompactionService implements ICompactionService {
  private llmProvider: LLMProvider;
  private contextService: ContextService;
  private logger: Logger;
  
  // 配置
  private readonly TRIGGER_RATIO = 0.9;        // 触发压缩比例
  private readonly MAX_OUTPUT_TOKENS = 4096;  // 摘要最大 Token
  private readonly KEEP_MESSAGES_NUM = 40;    // 保留最近消息数
  
  /**
   * 检查是否需要压缩
   */
  async shouldCompact(sessionId: string): Promise<boolean> {
    const tokenInfo = await this.getTokenInfo(sessionId);
    return tokenInfo.shouldCompact;
  }
  
  /**
   * 获取 Token 信息
   */
  async getTokenInfo(sessionId: string): Promise<TokenInfo> {
    const context = await this.contextService.getContext(sessionId);
    
    const messages = context.messages;
    const nonSystemMessages = messages.filter(m => m.role !== 'system');
    
    const estimatedTotal = this.estimateTokens(messages);
    const usableLimit = MAX_CONTEXT_TOKENS - MAX_OUTPUT_TOKENS;
    const threshold = usableLimit * TRIGGER_RATIO;
    
    return {
      estimatedTotal,
      usableLimit,
      threshold,
      shouldCompact: estimatedTotal >= threshold && nonSystemMessages.length > KEEP_MESSAGES_NUM,
      messageCount: messages.length,
    };
  }
  
  /**
   * 执行压缩
   * 
   * 使用分布式锁保证单次压缩
   */
  async compact(sessionId: string): Promise<CompactionResult> {
    // 1. 获取分布式锁
    const lock = await this.redis.acquireLock(
      `compact:${sessionId}`,
      30000  // 30秒超时
    );
    
    try {
      // 2. 二次检查 (获取锁后可能已被其他进程压缩)
      if (!await this.shouldCompact(sessionId)) {
        return { success: false };
      }
      
      // 3. 获取上下文
      const context = await this.contextService.getContext(sessionId);
      
      // 4. 分离消息区域
      const { systemMessage, pending, active } = this.splitMessages(
        context.messages
      );
      
      // 5. 处理工具调用配对
      const { pending: finalPending, active: finalActive } = 
        await this.processToolCallPairs(pending, active);
      
      // 6. 生成摘要
      const summary = await this.generateSummary(finalPending);
      
      // 7. 重组消息
      const newMessages = [
        systemMessage,
        summary,
        ...finalActive,
      ].filter(Boolean);
      
      // 8. 计算压缩后 Token
      const tokensAfter = this.estimateTokens(newMessages);
      
      // 9. 更新上下文
      await this.contextService.updateContext(sessionId, {
        messages: newMessages,
        stats: {
          totalTokens: tokensAfter,
          messageCount: newMessages.length,
          lastCompactionAt: new Date(),
        },
      });
      
      // 10. 创建压缩记录
      await this.createCompactionRecord(sessionId, {
        messagesBefore: context.messages.length,
        messagesAfter: newMessages.length,
        tokensBefore: this.estimateTokens(context.messages),
        tokensAfter,
        summaryMessageId: summary.id,
      });
      
      this.logger.info('Compaction completed', {
        sessionId,
        messagesBefore: context.messages.length,
        messagesAfter: newMessages.length,
      });
      
      return {
        success: true,
        summaryMessage: summary,
        messagesBefore: context.messages.length,
        messagesAfter: newMessages.length,
        tokensBefore: this.estimateTokens(context.messages),
        tokensAfter,
      };
      
    } finally {
      // 10. 释放锁
      await lock.release();
    }
  }
  
  /**
   * 分离消息区域
   */
  private splitMessages(messages: Message[]): {
    systemMessage: Message | undefined;
    pending: Message[];
    active: Message[];
  } {
    const systemMessage = messages.find(m => m.role === 'system');
    const nonSystem = messages.filter(m => m.role !== 'system');
    
    // 保留最近 N 条消息
    const splitPoint = Math.max(0, nonSystem.length - this.KEEP_MESSAGES_NUM);
    
    return {
      systemMessage,
      pending: nonSystem.slice(0, splitPoint),
      active: nonSystem.slice(splitPoint),
    };
  }
  
  /**
   * 处理工具调用配对
   * 
   * 确保 active 区的 tool 消息有对应的 assistant 消息
   */
  private async processToolCallPairs(
    pending: Message[],
    active: Message[]
  ): Promise<{ pending: Message[]; active: Message[] }> {
    // 构建 toolCallId -> assistant 消息的映射
    const toolCallMap = new Map<string, Message>();
    
    for (const msg of [...pending, ...active]) {
      if (msg.role === 'assistant' && msg.toolCalls) {
        for (const call of msg.toolCalls) {
          toolCallMap.set(call.id, msg);
        }
      }
    }
    
    // 找出 active 区需要配对的 tool 消息
    const activeTools = active.filter(m => m.role === 'tool');
    const toolsNeedingPair = activeTools.filter(toolMsg => {
      const callId = toolMsg.toolCallId;
      return callId && toolCallMap.has(callId);
    });
    
    if (toolsNeedingPair.length === 0) {
      return { pending, active };
    }
    
    // 需要将对应的 assistant 消息从 pending 移到 active
    const assistantsToMove = new Set<Message>();
    for (const toolMsg of toolsNeedingPair) {
      const assistantMsg = toolCallMap.get(toolMsg.toolCallId!);
      if (assistantMsg) {
        assistantsToMove.add(assistantMsg);
      }
    }
    
    // 过滤 pending
    const newPending = pending.filter(m => !assistantsToMove.has(m));
    
    // 重组 active
    const newActive: Message[] = [];
    for (const assistant of assistantsToMove) {
      newActive.push(assistant);
      // 添加对应的 tool 消息
      for (const call of assistant.toolCalls || []) {
        const toolMsg = active.find(
          m => m.role === 'tool' && m.toolCallId === call.id
        );
        if (toolMsg) {
          newActive.push(toolMsg);
        }
      }
    }
    // 添加剩余的 active 消息
    for (const msg of active) {
      if (!newActive.includes(msg)) {
        newActive.push(msg);
      }
    }
    
    return { pending: newPending, active: newActive };
  }
  
  /**
   * 生成摘要
   */
  private async generateSummary(messages: Message[]): Promise<Message> {
    if (messages.length === 0) {
      return {
        id: generateUUID(),
        sessionId: '',
        role: 'assistant',
        type: 'summary',
        content: 'No previous messages to summarize.',
        createdAt: new Date(),
      };
    }
    
    // 构建摘要提示
    const prompt = this.buildSummaryPrompt(messages);
    
    // 调用 LLM 生成摘要
    const response = await this.llmProvider.generate(
      [{ role: 'user', content: prompt }],
      { maxTokens: this.MAX_OUTPUT_TOKENS }
    );
    
    const content = this.extractContent(response);
    
    return {
      id: generateUUID(),
      sessionId: '',
      role: 'assistant',
      type: 'summary',
      content,
      createdAt: new Date(),
    };
  }
  
  /**
   * 构建摘要提示
   */
  private buildSummaryPrompt(messages: Message[]): string {
    const messageTexts = messages
      .map(m => `${m.role}: ${this.contentToText(m.content)}`)
      .join('\n\n');
    
    return `请将以下对话历史压缩成结构化摘要，包含：
1. 用户核心需求
2. 关键的技术决策
3. 修改的文件和代码片段
4. 遇到的错误和解决方案
5. 待完成的任务
6. 当前工作进度

对话历史：
${messageText}

要求：
- 使用中文
- 信息密度高
- 保持关键上下文
- 摘要应该能让后续对话保持连贯性`;
  }
  
  /**
   * 估算 Token 数量
   */
  private estimateTokens(messages: Message[]): number {
    const text = JSON.stringify(messages);
    
    let cnCount = 0;
    let otherCount = 0;
    
    for (const char of text) {
      if (char >= '\u4e00' && char <= '\u9fa5') {
        cnCount++;
      } else {
        otherCount++;
      }
    }
    
    // 中文约 1.5 token/字符，其他约 0.25 token/字符
    return Math.ceil(cnCount * 1.5 + otherCount * 0.25);
  }
  
  /**
   * 内容转文本
   */
  private contentToText(content: MessageContent): string {
    if (typeof content === 'string') {
      return content;
    }
    if (Array.isArray(content)) {
      return content
        .filter(p => p.type === 'text')
        .map(p => p.text)
        .join('');
    }
    return '';
  }
}
```

### 7.2 分布式锁实现

```typescript
/**
 * 分布式锁服务
 * 
 * 使用 Redis 实现 RedLock 算法
 */
class DistributedLock {
  private redis: Redis;
  
  /**
   * 获取锁
   */
  async acquireLock(
    resource: string, 
    ttl: number,
    options?: {
      retryInterval?: number;
      maxRetries?: number;
    }
  ): Promise<Lock> {
    const lockKey = `lock:${resource}`;
    const lockValue = generateUUID();
    
    const retryInterval = options?.retryInterval ?? 100;
    const maxRetries = options?.maxRetries ?? 10;
    
    for (let i = 0; i < maxRetries; i++) {
      // SET NX EX (原子操作)
      const result = await this.redis.set(lockKey, lockValue, {
        NX: true,
        EX: Math.ceil(ttl / 1000),
      });
      
      if (result === 'OK') {
        return new Lock(this.redis, lockKey, lockValue, ttl);
      }
      
      // 等待重试
      await this.sleep(retryInterval);
    }
    
    throw new LockAcquireError(`Failed to acquire lock: ${resource}`);
  }
  
  /**
   * 释放锁 (Lua 脚本保证原子性)
   */
  async releaseLock(lockKey: string, lockValue: string): Promise<void> {
    const script = `
      if redis.call("get", KEYS[1]) == ARGV[1] then
        return redis.call("del", KEYS[1])
      else
        return 0
      end
    `;
    
    await this.redis.eval(script, {
      keys: [lockKey],
      arguments: [lockValue],
    });
  }
}

/**
 * 锁对象
 */
class Lock {
  private released = false;
  
  constructor(
    private redis: Redis,
    private key: string,
    private value: string,
    private ttl: number
  ) {}
  
  /**
   * 释放锁
   */
  async release(): Promise<void> {
    if (this.released) {
      return;
    }
    
    await this.redis.eval(`
      if redis.call("get", KEYS[1]) == ARGV[1] then
        return redis.call("del", KEYS[1])
      else
        return 0
      end
    `, {
      keys: [this.key],
      arguments: [this.value],
    });
    
    this.released = true;
  }
  
  /**
   * 延长锁
   */
  async extend(additionalTtl: number): Promise<void> {
    if (this.released) {
      throw new Error('Lock already released');
    }
    
    const script = `
      if redis.call("get", KEYS[1]) == ARGV[1] then
        return redis.call("pexpire", KEYS[1], ARGV[2])
      else
        return 0
      end
    `;
    
    const result = await this.redis.eval(script, {
      keys: [this.key],
      arguments: [this.value, this.ttl + additionalTtl],
    });
    
    if (result !== 1) {
      throw new Error('Failed to extend lock');
    }
    
    this.ttl += additionalTtl;
  }
}
```
