### 1.2 需求目标

| 需求 | 描述 | 优先级 |
|------|------|--------|
| **后台执行** | 关闭浏览器/页面后，任务继续在服务端执行 | P0 |
| **状态持久化** | 任务状态定期保存，支持进程重启恢复 | P0 |
| **进度实时通知** | 任务执行过程中，通过 WebSocket 推送进度 | P1 |
| **任务管理** | 支持查询、取消、恢复任务 | P1 |
| **多任务并发** | 支持多个后台任务同时执行 | P2 |
| **大规模并发** | 支持万级用户同时使用 (如 MiniMax 平台) | P1 |

## 3. 数据模型

### 3.1 任务表 (tasks)

```sql
CREATE TABLE tasks (
    -- 核心字段
    id              VARCHAR(64) PRIMARY KEY,  -- task_时间戳_随机串
    type            VARCHAR(32) NOT NULL,      -- 'agent' | 'compact' | 'scheduled'
    status          VARCHAR(32) NOT NULL DEFAULT 'pending',
    
    -- 关联
    user_id         UUID NOT NULL,
    tenant_id       UUID NOT NULL,
    session_id      VARCHAR(64),               -- 关联的会话 ID (可选)
    
    -- 任务内容
    payload         JSONB NOT NULL,            -- 任务参数
    priority        INTEGER DEFAULT 0,         -- 优先级 (越大越高)
    
    -- 执行信息
    worker_id       VARCHAR(64),                -- 执行该任务的 Worker ID
    started_at      TIMESTAMP WITH TIME ZONE,  -- 开始时间
    finished_at     TIMESTAMP WITH TIME ZONE,  -- 结束时间
    progress        INTEGER DEFAULT 0,          -- 进度百分比 0-100
    
    -- 结果
    result          JSONB,                      -- 执行结果
    error           TEXT,                       -- 错误信息
    
    -- 元数据
    created_at      TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at      TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    
    -- 索引
    CONSTRAINT tasks_status CHECK (status IN (
        'pending', 'queued', 'running', 'paused', 
        'completed', 'failed', 'cancelled'
    ))
);

-- 索引
CREATE INDEX idx_tasks_user_status ON tasks(user_id, status);
CREATE INDEX idx_tasks_tenant_status ON tasks(tenant_id, status);
CREATE INDEX idx_tasks_session ON tasks(session_id);
CREATE INDEX idx_tasks_priority ON tasks(priority DESC, created_at ASC);
CREATE INDEX idx_tasks_created_at ON tasks(created_at DESC);
```

### 3.2 任务历史表 (task_logs)

```sql
CREATE TABLE task_logs (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    task_id         VARCHAR(64) NOT NULL REFERENCES tasks(id) ON DELETE CASCADE,
    level           VARCHAR(16) NOT NULL,     -- 'info' | 'warning' | 'error'
    message         TEXT NOT NULL,
    data            JSONB,                     -- 额外数据
    created_at      TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
```

## 5. 核心模块设计

### 5.1 任务调度器 (TaskScheduler)

```typescript
/**
 * 任务调度器
 * 
 * 职责：
 * 1. 接收任务请求
 * 2. 创建任务记录
 * 3. 入队任务
 * 4. 管理任务状态
 */
class TaskScheduler {
    private taskQueue: TaskQueue;
    private taskStore: TaskStore;
    private webSocketManager: WebSocketManager;
    
    /**
     * 创建后台任务
     */
    async createTask(params: CreateTaskParams): Promise<Task> {
        // 1. 验证用户权限
        await this.authorize(params.userId, params.tenantId, 'task:create');
        
        // 2. 创建任务记录
        const task = await this.taskStore.create({
            id: this.generateTaskId(),
            type: params.type,
            userId: params.userId,
            tenantId: params.tenantId,
            sessionId: params.sessionId,
            payload: params.payload,
            priority: params.priority || 0,
            status: 'pending',
        });
        
        // 3. 加入任务队列
        await this.taskQueue.push({
            taskId: task.id,
            type: task.type,
            payload: task.payload,
            priority: task.priority,
        });
        
        // 4. 更新状态为 queued
        await this.taskStore.update(task.id, { status: 'queued' });
        
        // 5. WebSocket 通知
        this.webSocketManager.publish(`user:${params.userId}`, {
            event: 'task:created',
            data: task,
        });
        
        return task;
    }
    
    /**
     * 取消任务
     */
    async cancelTask(taskId: string, userId: string): Promise<Task> {
        const task = await this.taskStore.findById(taskId);
        
        if (!task) {
            throw new TaskNotFoundError(taskId);
        }
        
        // 权限检查
        if (task.userId !== userId) {
            throw new ForbiddenError();
        }
        
        // 只能取消 pending/queued/running 状态的任务
        if (!['pending', 'queued', 'running'].includes(task.status)) {
            throw new InvalidTaskStatusError(task.status, ['pending', 'queued', 'running']);
        }
        
        // 更新状态
        const updated = await this.taskStore.update(taskId, {
            status: 'cancelled',
            finishedAt: new Date(),
        });
        
        // 通知 Worker 停止
        if (task.status === 'running' && task.workerId) {
            await this.notifyWorkerToStop(task.workerId, taskId);
        }
        
        return updated;
    }
    
    /**
     * 生成任务 ID
     */
    private generateTaskId(): string {
        const timestamp = Date.now();
        const random = Math.random().toString(36).substring(2, 8);
        return `task_${timestamp}_${random}`;
    }
}
```

### 5.2 任务执行器 (TaskExecutor)

```typescript
/**
 * 任务执行器
 * 
 * 职责：
 * 1. 从队列获取任务
 * 2. 执行任务
 * 3. 报告进度
 * 4. 处理错误
 */
class TaskExecutor {
    private taskStore: TaskStore;
    private agentFactory: AgentFactory;
    private heartbeatInterval = 5000;  // 5 秒心跳
    
    /**
     * 执行单个任务
     */
    async executeTask(task: Task): Promise<void> {
        const workerId = this.getWorkerId();
        
        try {
            // 1. 更新状态为 running
            await this.taskStore.update(task.id, {
                status: 'running',
                workerId,
                startedAt: new Date(),
            });
            
            // 2. 创建 Agent 实例
            const agent = await this.agentFactory.create(task.payload);
            
            // 3. 启动心跳
            const heartbeatTimer = this.startHeartbeat(task.id, agent);
            
            // 4. 执行任务
            const result = await agent.execute(task.payload.prompt);
            
            // 5. 停止心跳
            clearInterval(heartbeatTimer);
            
            // 6. 保存结果
            await this.taskStore.update(task.id, {
                status: 'completed',
                finishedAt: new Date(),
                progress: 100,
                result: {
                    output: result.output,
                    turns: result.turns,
                    toolsUsed: result.toolsUsed,
                },
            });
            
        } catch (error) {
            // 处理错误
            await this.handleError(task, error);
        }
    }
    
    /**
     * 启动心跳
     */
    private startHeartbeat(taskId: string, agent: Agent): NodeJS.Timeout {
        return setInterval(async () => {
            try {
                // 1. 获取 Agent 进度
                const progress = agent.getProgress();
                
                // 2. 保存快照
                await this.saveSnapshot(taskId, {
                    messages: agent.getMessages(),
                    agentState: agent.getState(),
                });
                
                // 3. 更新进度
                await this.taskStore.update(taskId, {
                    progress: progress.percent,
                    updatedAt: new Date(),
                });
                
                // 4. 发布 WebSocket 事件
                await this.publishProgress(taskId, progress);
                
            } catch (error) {
                console.error(`Heartbeat failed for task ${taskId}:`, error);
            }
        }, this.heartbeatInterval);
    }
    
    /**
     * 保存进度快照
     */
    private async saveSnapshot(taskId: string, data: SnapshotData): Promise<void> {
        // 限制保存频率，避免存储爆炸
        const lastSnapshot = await this.taskStore.getLastSnapshot(taskId);
        if (lastSnapshot) {
            const elapsed = Date.now() - new Date(lastSnapshot.createdAt).getTime();
            if (elapsed < 30000) {  // 至少 30 秒间隔
                return;
            }
        }
        
        await this.taskStore.createSnapshot({
            taskId,
            messages: data.messages,
            agentState: data.agentState,
            snapshotType: 'heartbeat',
        });
    }
}
```

### 5.3 Worker 池 (Worker Pool)

```typescript
/**
 * Worker 池
 * 
 * 职责：
 * 1. 管理多个 TaskExecutor
 * 2. 控制并发数
 * 3. 负载均衡
 */
class WorkerPool {
    private workers: Map<string, TaskExecutor> = new Map();
    private queue: TaskQueue;
    private maxWorkers: number;
    private runningCount = 0;
    
    constructor(queue: TaskQueue, maxWorkers: number = 5) {
        this.queue = queue;
        this.maxWorkers = maxWorkers;
    }
    
    /**
     * 启动 Worker 池
     */
    async start(): Promise<void> {
        // 启动多个 Worker
        for (let i = 0; i < this.maxWorkers; i++) {
            this.spawnWorker(`worker-${i}`);
        }
        
        // 监听新任务
        this.queue.on('task', async (task) => {
            await this.dispatchTask(task);
        });
    }
    
    /**
     * 分发任务
     */
    private async dispatchTask(task: Task): Promise<void> {
        // 等待可用的 Worker
        while (this.runningCount >= this.maxWorkers) {
            await this.waitForAvailableWorker();
        }
        
        // 分配任务
        const worker = this.getAvailableWorker();
        if (!worker) {
            throw new Error('No available worker');
        }
        
        this.runningCount++;
        try {
            await worker.executeTask(task);
        } finally {
            this.runningCount--;
        }
    }
    
    /**
     * 生成 Worker
     */
    private spawnWorker(workerId: string): void {
        const executor = new TaskExecutor(/* ... */);
        this.workers.set(workerId, executor);
    }
}
```

```typescript
/**
 * 任务进度
 */
interface TaskProgress {
    taskId: string;
    status: TaskStatus;
    progress: number;        // 0-100
    currentStep?: string;   // 当前步骤描述
    currentTool?: string;   // 当前执行的工具
    message?: string;       // 最新消息
    turns: number;          // 已执行轮次
    toolsUsed: string[];    // 已使用的工具
}

/**
 * 进度计算策略
 */
class ProgressCalculator {
    /**
     * 基于 Agent 步骤计算进度
     */
    calculateBySteps(currentStep: number, maxSteps: number): number {
        return Math.min(100, Math.round((currentStep / maxSteps) * 100));
    }
    
    /**
     * 基于工具调用计算进度
     */
    calculateByTools(toolsUsed: string[], estimatedTools: string[]): number {
        const used = new Set(toolsUsed);
        const estimated = new Set(estimatedTools);
        
        // 简单估算：已使用工具数 / 预估工具数
        const ratio = used.size / Math.max(estimated.size, 1);
        return Math.min(90, Math.round(ratio * 100));  // 保留 10% 给最后处理
    }
    
    /**
     * 基于消息数计算进度
     */
    calculateByMessages(currentMessages: number, threshold: number): number {
        return Math.min(100, Math.round((currentMessages / threshold) * 100));
    }
}
```

### 6.3 WebSocket 推送

```typescript
/**
 * WebSocket 管理器
 */
class WebSocketManager {
    private connections: Map<string, Set<WebSocket>> = new Map();
    
    /**
     * 订阅用户任务事件
     */
    subscribe(userId: string, ws: WebSocket): void {
        const key = `user:${userId}`;
        if (!this.connections.has(key)) {
            this.connections.set(key, new Set());
        }
        this.connections.get(key)!.add(ws);
    }
    
    /**
     * 发布任务事件
     */
    async publish(userId: string, event: TaskEvent): Promise<void> {
        const key = `user:${userId}`;
        const sockets = this.connections.get(key);
        if (!sockets) return;
        
        const message = JSON.stringify(event);
        for (const ws of sockets) {
            if (ws.readyState === WebSocket.OPEN) {
                ws.send(message);
            }
        }
    }
    
    /**
     * 广播任务更新
     */
    async broadcastTaskUpdate(task: Task): Promise<void> {
        await this.publish(task.userId, {
            event: `task:${task.status}`,
            data: {
                taskId: task.id,
                status: task.status,
                progress: task.progress,
            },
        });
    }
}
```

```typescript
/**
 * 任务恢复服务
 */
class TaskRecoveryService {
    private taskStore: TaskStore;
    private taskQueue: TaskQueue;
    private staleThresholdMs = 30000;  // 30 秒超时
    
    /**
     * 检测并恢复 stale 任务
     */
    async detectAndRecoverStaleTasks(): Promise<number> {
        // 1. 查找 stale 任务
        const staleTasks = await this.taskStore.findStaleTasks({
            threshold: this.staleThresholdMs,
            statuses: ['running'],
        });
        
        // 2. 逐个恢复
        let recovered = 0;
        for (const task of staleTasks) {
            const success = await this.recoverTask(task);
            if (success) recovered++;
        }
        
        return recovered;
    }
    
    /**
     * 恢复单个任务
     */
    private async recoverTask(task: Task): Promise<boolean> {
        try {
            // 1. 获取最新快照
            const snapshot = await this.taskStore.getLastSnapshot(task.id);
            if (!snapshot) {
                // 无快照，标记为失败
                await this.taskStore.update(task.id, {
                    status: 'failed',
                    error: 'No snapshot available for recovery',
                    finishedAt: new Date(),
                });
                return false;
            }
            
            // 2. 从快照恢复 Agent 状态
            const agent = Agent.fromSnapshot(snapshot.agentState);
            
            // 3. 更新状态为 running
            await this.taskStore.update(task.id, {
                status: 'running',
                workerId: null,  // 清除旧的 Worker
            });
            
            // 4. 重新入队（由新的 Worker 领取执行）
            await this.taskQueue.push({
                taskId: task.id,
                type: task.type,
                payload: task.payload,
                priority: task.priority,
                resumeFromSnapshot: snapshot.id,
            });
            
            return true;
            
        } catch (error) {
            console.error(`Failed to recover task ${task.id}:`, error);
            await this.taskStore.update(task.id, {
                status: 'failed',
                error: `Recovery failed: ${error.message}`,
                finishedAt: new Date(),
            });
            return false;
        }
    }
}
```

```typescript
/**
 * 任务权限检查中间件
 */
async function checkTaskPermission(
    userId: string,
    tenantId: string,
    taskId: string,
    action: 'read' | 'update' | 'delete' | 'cancel'
): Promise<void> {
    const task = await taskStore.findById(taskId);
    if (!task) {
        throw new TaskNotFoundError(taskId);
    }
    
    // 超级管理员可以操作所有任务
    if (await hasPermission(userId, 'admin:all')) {
        return;
    }
    
    // 租户管理员可以操作租户内的所有任务
    if (action === 'read') {
        if (task.tenantId === tenantId && await hasPermission(userId, 'tenant:task:read')) {
            return;
        }
    }
    
    // 普通用户只能操作自己的任务
    if (task.userId === userId) {
        return;
    }
    
    throw new ForbiddenError(`Cannot ${action} task ${taskId}`);
}
```

### 13.1 任务队列 - Redis 实现

任务队列是整个后台任务系统的核心，它实现了**生产者-消费者模式**，解耦了任务提交和任务执行。

```typescript
// src/background-tasks/queue/task-queue.ts

import Redis from 'ioredis';
import { logger } from '../../utils/logger';

export interface TaskMessage {
  taskId: string;
  type: 'agent' | 'compact' | 'scheduled';
  payload: any;
  priority: number;
  createdAt: number;
  retryCount?: number;
}

export class TaskQueue {
  private redis: Redis;
  private readonly MAIN_QUEUE_KEY = 'agent:task:queue:main';
  private readonly DELAYED_QUEUE_KEY = 'agent:task:queue:delayed';
  private readonly PROCESSING_SET_KEY = 'agent:task:processing';
  
  constructor(redis: Redis) {
    this.redis = redis;
  }
  
  /**
   * 添加任务到队列 (生产者调用)
   * 
   * 使用 Redis List 的 LPUSH 从左边插入
   * BRPOP 会从右边取出，保证 FIFO 顺序
   */
  async enqueue(task: TaskMessage): Promise<void> {
    const queueKey = this.getQueueKey(task.type);
    
    // 添加任务到队列
    await this.redis.lpush(queueKey, JSON.stringify(task));
    
    // 设置队列过期时间 (24小时)
    await this.redis.expire(queueKey, 86400);
    
    logger.info(`[TaskQueue] 任务 ${task.taskId} 已入队`, {
      type: task.type,
      priority: task.priority,
    });
  }
  
  /**
   * 添加延迟任务 (用于重试)
   * 
   * 使用 Redis Sorted Set 实现延迟队列
   * score = 执行时间戳
   */
  async enqueueDelayed(task: TaskMessage, delayMs: number): Promise<void> {
    const executeAt = Date.now() + delayMs;
    
    await this.redis.zadd(
      this.DELAYED_QUEUE_KEY,
      executeAt,
      JSON.stringify(task)
    );
    
    logger.info(`[TaskQueue] 任务 ${task.taskId} 已加入延迟队列`, {
      delayMs,
      executeAt: new Date(executeAt).toISOString(),
    });
  }
  
  /**
   * 从队列取出任务 (Worker 调用)
   * 
   * BRPOP 是阻塞等待版本：
   * - 如果队列有任务，立即返回
   * - 如果队列为空，会一直等待，直到有任务或超时
   * - timeout = 0 表示无限等待
   */
  async dequeue(timeoutSeconds: number = 0): Promise<TaskMessage | null> {
    // 1. 先检查延迟队列是否有需要执行的任务
    const now = Date.now();
    const delayedTasks = await this.redis.zrangebyscore(
      this.DELAYED_QUEUE_KEY,
      0,
      now
    );
    
    if (delayedTasks.length > 0) {
      // 将到期的延迟任务移入主队列
      for (const taskData of delayedTasks) {
        const task = JSON.parse(taskData) as TaskMessage;
        await this.redis.zrem(this.DELAYED_QUEUE_KEY, taskData);
        await this.enqueue(task);
        logger.info(`[TaskQueue] 延迟任务 ${task.taskId} 已转入主队列`);
      }
    }
    
    // 2. 从主队列获取任务 (阻塞等待)
    const result = await this.redis.brpop(this.MAIN_QUEUE_KEY, timeoutSeconds);
    
    if (!result) {
      return null;  // 超时或队列为空
    }
    
    // result[0] = key, result[1] = 数据
    const [, data] = result;
    const task = JSON.parse(data) as TaskMessage;
    
    // 3. 将任务加入处理中集合 (防止重复处理)
    await this.redis.sadd(this.PROCESSING_SET_KEY, task.taskId);
    
    logger.info(`[TaskQueue] 任务 ${task.taskId} 已出队`);
    return task;
  }
  
  /**
   * 完成任务，移出处理中集合
   */
  async complete(taskId: string): Promise<void> {
    await this.redis.srem(this.PROCESSING_SET_KEY, taskId);
    logger.info(`[TaskQueue] 任务 ${taskId} 已完成`);
  }
  
  /**
   * 任务失败，重新入队 (带延迟)
   */
  async requeue(task: TaskMessage, maxRetries: number = 3): Promise<void> {
    // 移出处理中集合
    await this.redis.srem(this.PROCESSING_SET_KEY, task.taskId);
    
    // 检查重试次数
    const retryCount = (task.retryCount || 0) + 1;
    if (retryCount > maxRetries) {
      logger.error(`[TaskQueue] 任务 ${task.taskId} 超过最大重试次数`, {
        retryCount,
        maxRetries,
      });
      return;  // 不再重试
    }
    
    // 计算延迟 (指数退避: 1s, 2s, 4s, 8s...)
    const delayMs = Math.pow(2, retryCount - 1) * 1000;
    
    // 更新重试次数
    task.retryCount = retryCount;
    
    // 加入延迟队列
    await this.enqueueDelayed(task, delayMs);
    
    logger.warn(`[TaskQueue] 任务 ${task.taskId} 重新入队`, {
      retryCount,
      delayMs,
    });
  }
  
  /**
   * 获取队列长度
   */
  async size(): Promise<number> {
    return this.redis.llen(this.MAIN_QUEUE_KEY);
  }
  
  /**
   * 获取延迟队列长度
   */
  async delayedSize(): Promise<number> {
    return this.redis.zcard(this.DELAYED_QUEUE_KEY);
  }
  
  /**
   * 获取指定类型的队列键
   */
  private getQueueKey(type: string): string {
    return `${this.MAIN_QUEUE_KEY}:${type}`;
  }
}
```

### 13.2 Worker 进程 - 持续运行的任务执行器

Worker 是独立运行的进程，它的生命周期与客户端完全无关。

```typescript
// src/background-tasks/worker/worker.ts

import { TaskQueue, TaskMessage } from '../queue/task-queue';
import { TaskStore } from '../store/task-store';
import { AgentExecutor } from './agent-executor';
import { logger } from '../../utils/logger';
import { Redis } from 'ioredis';

export class Worker {
  private queue: TaskQueue;
  private taskStore: TaskStore;
  private agentExecutor: AgentExecutor;
  private workerId: string;
  private running = false;
  private redis: Redis;
  
  constructor() {
    // 初始化 Redis 连接
    this.redis = new Redis(process.env.REDIS_URL || 'redis://localhost:6379');
    
    // 初始化组件
    this.queue = new TaskQueue(this.redis);
    this.taskStore = new TaskStore();
    this.agentExecutor = new AgentExecutor();
    
    // 生成唯一 Worker ID
    this.workerId = `worker-${process.env.HOSTNAME || 'local'}-${process.pid}-${Date.now()}`;
  }
  
  /**
   * Worker 启动入口
   * 
   * 启动后永远不会退出（除非进程被杀死）
   * 这是整个后台任务系统的核心
   */
  async start(): Promise<void> {
    this.running = true;
    
    logger.info(`[Worker] ${this.workerId} 启动成功`);
    
    // 注册 Worker 到 Redis (用于监控)
    await this.register();
    
    // 启动主循环
    await this.runLoop();
  }
  
  /**
   * 主循环 - 持续等待并执行任务
   */
  private async runLoop(): Promise<void> {
    while (this.running) {
      try {
        logger.debug(`[Worker] ${this.workerId} 等待新任务...`);
        
        // 从队列获取任务 (阻塞等待，有任务才返回)
        const task = await this.queue.dequeue(30);  // 30秒超时
        
        if (!task) {
          continue;  // 超时，继续等待
        }
        
        // 执行任务
        await this.executeTask(task);
        
      } catch (error) {
        logger.error(`[Worker] ${this.workerId} 执行出错:`, error);
        // 发生错误后继续等待下一个任务，不会退出
      }
    }
  }
  
  /**
   * 执行单个任务
   */
  private async executeTask(task: TaskMessage): Promise<void> {
    const { taskId, type, payload } = task;
    
    logger.info(`[Worker] ${this.workerId} 开始执行任务 ${taskId}`, {
      type,
    });
    
    try {
      // === 步骤 1: 更新任务状态为 "运行中" ===
      await this.taskStore.update(taskId, {
        status: 'running',
        workerId: this.workerId,
        startedAt: new Date(),
      });
      
      // === 步骤 2: 根据任务类型执行 ===
      let result: any;
      
      if (type === 'agent') {
        // 执行 Agent 任务
        result = await this.agentExecutor.execute(taskId, payload);
        
      } else if (type === 'compact') {
        // 执行压缩任务
        result = await this.executeCompaction(payload);
        
      } else if (type === 'scheduled') {
        // 执行定时任务
        result = await this.executeScheduledTask(payload);
        
      } else {
        throw new Error(`Unknown task type: ${type}`);
      }
      
      // === 步骤 3: 任务完成，保存结果 ===
      await this.taskStore.update(taskId, {
        status: 'completed',
        finishedAt: new Date(),
        progress: 100,
        result: result,
      });
      
      // === 步骤 4: 从处理中集合移除 ===
      await this.queue.complete(taskId);
      
      logger.info(`[Worker] 任务 ${taskId} 执行完成`, {
        workerId: this.workerId,
      });
      
    } catch (error: any) {
      // === 步骤 5: 任务失败 ===
      logger.error(`[Worker] 任务 ${taskId} 执行失败:`, {
        error: error.message,
        stack: error.stack,
      });
      
      // 尝试重新入队
      await this.queue.requeue(task);
      
      // 更新数据库状态
      await this.taskStore.update(taskId, {
        status: 'failed',
        finishedAt: new Date(),
        error: error.message,
      });
    }
  }
  
  /**
   * 执行上下文压缩任务
   */
  private async executeCompaction(payload: any): Promise<any> {
    const { sessionId } = payload;
    
    // 调用压缩服务
    const compactionService = new CompactionService();
    const result = await compactionService.compact(sessionId);
    
    return result;
  }
  
  /**
   * 执行定时任务
   */
  private async executeScheduledTask(payload: any): Promise<any> {
    const { action, params } = payload;
    
    switch (action) {
      case 'cleanup':
        // 清理过期会话
        return await this.cleanupExpiredSessions();
        
      case 'compaction':
        // 压缩所有需要压缩的会话
        return await this.compactAllNeeded();
        
      default:
        throw new Error(`Unknown scheduled action: ${action}`);
    }
  }
  
  /**
   * 注册 Worker 到 Redis
   */
  private async register(): Promise<void> {
    const key = `agent:workers:${this.workerId}`;
    await this.redis.setex(key, 300, JSON.stringify({
      startedAt: new Date().toISOString(),
      status: 'running',
    }));
    
    // 定期更新心跳
    setInterval(async () => {
      await this.redis.setex(key, 300, JSON.stringify({
        lastHeartbeat: new Date().toISOString(),
        status: 'running',
      }));
    }, 60000);  // 每分钟更新
  }
  
  /**
   * 停止 Worker
   */
  async stop(): Promise<void> {
    this.running = false;
    logger.info(`[Worker] ${this.workerId} 已停止`);
  }
}

// ==================== Worker 启动脚本 ====================

// src/background-tasks/worker/index.ts

async function main() {
  const worker = new Worker();
  
  // 处理进程信号
  process.on('SIGTERM', async () => {
    logger.info('收到 SIGTERM 信号，正在停止 Worker...');
    await worker.stop();
    process.exit(0);
  });
  
  process.on('SIGINT', async () => {
    logger.info('收到 SIGINT 信号，正在停止 Worker...');
    await worker.stop();
    process.exit(0);
  });
  
  // 启动 Worker
  await worker.start();
}

// 运行
main().catch((error) => {
  logger.error('Worker 启动失败:', error);
  process.exit(1);
});
```

### 13.3 API 接口 - 任务提交

```typescript
// src/routes/task-routes.ts

import { FastifyInstance, FastifyRequest, FastifyReply } from 'fastify';
import { TaskQueue } from '../background-tasks/queue/task-queue';
import { TaskStore } from '../background-tasks/store/task-store';
import { z } from 'zod';
import { logger } from '../utils/logger';

// 请求验证 Schema
const CreateTaskSchema = z.object({
  type: z.enum(['agent', 'compact', 'scheduled']).default('agent'),
  sessionId: z.string().optional(),
  payload: z.object({
    prompt: z.string().optional(),
    model: z.string().optional(),
    tools: z.array(z.string()).optional(),
    maxSteps: z.number().optional(),
    options: z.record(z.any()).optional(),
  }),
  priority: z.number().default(0),
});

type CreateTaskBody = z.infer<typeof CreateTaskSchema>;

export async function taskRoutes(fastify: FastifyInstance) {
  const queue = new TaskQueue(fastify.redis);
  const taskStore = new TaskStore();
  
  /**
   * POST /v1/tasks - 创建后台任务
   * 
   * 这个接口只是把任务加入队列，立即返回
   * 实际的执行由 Worker 在后台完成
   * 
   * 关键点：任务提交与任务执行是解耦的
   */
  fastify.post<{ Body: CreateTaskBody }>(
    '/v1/tasks',
    {
      preHandler: [fastify.authenticate],  // 需要认证
    },
    async (request: FastifyRequest<{ Body: CreateTaskBody }>, reply: FastifyReply) => {
      const body = CreateTaskSchema.parse(request.body);
      const userId = request.user.id;
      const tenantId = request.user.tenantId;
      
      // 1. 生成唯一任务 ID
      const taskId = `task_${Date.now()}_${Math.random().toString(36).slice(2, 8)}`;
      
      logger.info(`[API] 创建任务 ${taskId}`, {
        type: body.type,
        userId,
      });
      
      // 2. 在数据库中创建任务记录 (状态: pending)
      await taskStore.create({
        id: taskId,
        type: body.type,
        status: 'pending',
        userId,
        tenantId,
        sessionId: body.sessionId,
        payload: body.payload,
        priority: body.priority,
        progress: 0,
      });
      
      // 3. 把任务加入 Redis 队列 (生产者)
      await queue.enqueue({
        taskId,
        type: body.type,
        payload: body.payload,
        priority: body.priority,
        createdAt: Date.now(),
      });
      
      // 4. 更新状态为 queued
      await taskStore.update(taskId, { status: 'queued' });
      
      // 5. 立即返回 (不等待任务执行完成)
      // HTTP 202 Accepted 表示请求已接收，正在异步处理
      return reply.code(202).send({
        success: true,
        data: {
          taskId,
          status: 'queued',
          message: '任务已提交，将在后台执行',
        },
      });
    }
  );
  
  /**
   * GET /v1/tasks - 获取任务列表
   */
  fastify.get(
    '/v1/tasks',
    {
      preHandler: [fastify.authenticate],
    },
    async (request, reply) => {
      const { status, limit = 20, offset = 0 } = request.query as any;
      const userId = request.user.id;
      const tenantId = request.user.tenantId;
      
      const result = await taskStore.findMany({
        userId,
        tenantId,
        status,
        limit: Math.min(limit, 100),
        offset,
      });
      
      return {
        success: true,
        data: {
          items: result.items,
          pagination: {
            total: result.total,
            limit,
            offset,
            hasMore: offset + limit < result.total,
          },
        },
      };
    }
  );
  
  /**
   * GET /v1/tasks/:taskId - 获取任务详情
   */
  fastify.get<{ Params: { taskId: string } }>(
    '/v1/tasks/:taskId',
    {
      preHandler: [fastify.authenticate],
    },
    async (request, reply) => {
      const { taskId } = request.params;
      const userId = request.user.id;
      
      const task = await taskStore.findById(taskId);
      
      if (!task) {
        return reply.code(404).send({
          success: false,
          error: { code: 'NOT_FOUND', message: 'Task not found' },
        });
      }
      
      // 权限检查
      if (task.userId !== userId && !request.user.isAdmin) {
        return reply.code(403).send({
          success: false,
          error: { code: 'FORBIDDEN', message: 'Access denied' },
        });
      }
      
      return {
        success: true,
        data: task,
      };
    }
  );
  
  /**
   * GET /v1/tasks/:taskId/output - 获取任务输出
   */
  fastify.get<{ Params: { taskId: string } }>(
    '/v1/tasks/:taskId/output',
    {
      preHandler: [fastify.authenticate],
    },
    async (request, reply) => {
      const { taskId } = request.params;
      
      const task = await taskStore.findById(taskId);
      
      if (!task) {
        return reply.code(404).send({
          success: false,
          error: { code: 'NOT_FOUND', message: 'Task not found' },
        });
      }
      
      // 只有完成的任务才能获取输出
      if (task.status !== 'completed') {
        return reply.code(400).send({
          success: false,
          error: { 
            code: 'INVALID_STATUS', 
            message: `Task is ${task.status}, output not available` 
          },
        });
      }
      
      return {
        success: true,
        data: {
          taskId: task.id,
          status: task.status,
          output: task.result?.output,
          metrics: {
            turns: task.result?.turns,
            toolsUsed: task.result?.toolsUsed,
            durationMs: task.result?.durationMs,
          },
        },
      };
    }
  );
  
  /**
   * DELETE /v1/tasks/:taskId - 取消任务
   */
  fastify.delete<{ Params: { taskId: string } }>(
    '/v1/tasks/:taskId',
    {
      preHandler: [fastify.authenticate],
    },
    async (request, reply) => {
      const { taskId } = request.params;
      const userId = request.user.id;
      
      const task = await taskStore.findById(taskId);
      
      if (!task) {
        return reply.code(404).send({
          success: false,
          error: { code: 'NOT_FOUND', message: 'Task not found' },
        });
      }
      
      // 权限检查
      if (task.userId !== userId && !request.user.isAdmin) {
        return reply.code(403).send({
          success: false,
          error: { code: 'FORBIDDEN', message: 'Access denied' },
        });
      }
      
      // 只能取消 pending/queued/running 状态的任务
      if (!['pending', 'queued', 'running'].includes(task.status)) {
        return reply.code(400).send({
          success: false,
          error: { 
            code: 'INVALID_STATUS', 
            message: `Cannot cancel task with status ${task.status}` 
          },
        });
      }
      
      // 更新状态
      await taskStore.update(taskId, {
        status: 'cancelled',
        finishedAt: new Date(),
      });
      
      return {
        success: true,
        data: {
          taskId,
          status: 'cancelled',
        },
      };
    }
  );
}
```

## 14. 总结

### 14.1 核心设计要点

| 要点 | 实现方式 |
|------|----------|
| **任务与页面解耦** | 任务提交后立即入队，与 HTTP 连接无关 |
| **Worker 常驻** | `while(true)` 循环 + `brpop` 阻塞等待 |
| **状态持久化** | 定期保存快照到 PostgreSQL |
| **故障恢复** | 心跳检测 + 快照恢复 |
| **并发控制** | 启动多个 Worker 进程 |

### 14.3 常见问题

**Q: Worker 进程如何保证不重复执行任务？**

A: 使用 Redis Set (`agent:task:processing`) 记录正在执行的任务，出队时加入，完成时移除。

**Q: 如果 Worker 崩溃正在执行的任务怎么办？**

A: 通过心跳检测发现后，从最新快照恢复任务，重新入队由其他 Worker 执行。

**Q: 如何控制并发数量？**

A: 通过启动的 Worker 进程数量控制，每个 Worker 同时只能执行一个任务。
