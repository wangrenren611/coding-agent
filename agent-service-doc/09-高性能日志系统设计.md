## 1. 概述

### 1.1 设计背景

现有的 `agent-v2` 日志系统已经具备了企业级日志的基础能力，包括：

- 多 Transport 支持（Console、File）
- 结构化日志
- 中间件机制
- 日志轮转
- 敏感字段脱敏

但在**大规模分布式系统**中，我们需要更强的能力：

| 需求 | 描述 |
|------|------|
| **分布式追踪** | 全链路追踪请求 |
| **高吞吐** | 支持百万级日志/秒 |
| **多目标输出** | Elasticsearch、Loki、Syslog 等 |
| **采样策略** | 智能采样，减少存储成本 |
| **告警集成** | 实时告警通知 |
| **日志分析** | 聚合统计、趋势分析 |

### 1.2 设计目标

本设计文档基于以下现状进行扩展：

**现有实现 (src/agent-v2/logger)**：
- 已实现：Console/File Transport、结构化日志、中间件机制、日志轮转、敏感字段脱敏

### 3.1 分布式追踪

```typescript
/**
 * Trace 上下文
 * 
 * 包含 TraceID 和 SpanID，用于全链路追踪
 */
interface TraceContext {
  traceId: string;      // 全局唯一追踪ID
  spanId: string;       // 当前 Span ID
  parentSpanId?: string; // 父 Span ID
  sampled: boolean;      // 是否采样
}

/**
 * Trace 管理器
 */
class TraceManager {
  private static instance: TraceManager;
  private currentContext: TraceContext | null = null;
  
  /**
   * 开始新的 Trace
   */
  startTrace(options?: { parent?: TraceContext }): TraceContext {
    const traceId = this.generateTraceId();
    const spanId = this.generateSpanId();
    
    const context: TraceContext = {
      traceId,
      spanId,
      parentSpanId: options?.parent?.spanId,
      sampled: this.shouldSample(),
    };
    
    this.currentContext = context;
    return context;
  }
  
  /**
   * 开始新的 Span
   */
  startSpan(name: string): TraceContext {
    const parent = this.currentContext;
    if (!parent) {
      return this.startTrace();
    }
    
    const spanId = this.generateSpanId();
    
    const context: TraceContext = {
      traceId: parent.traceId,
      spanId,
      parentSpanId: parent.spanId,
      sampled: parent.sampled,
    };
    
    this.currentContext = context;
    return context;
  }
  
  /**
   * 结束当前 Span
   */
  endSpan(): void {
    if (this.currentContext?.parentSpanId) {
      // 恢复到父 Span
      this.currentContext = {
        ...this.currentContext,
        spanId: this.currentContext.parentSpanId,
      };
    } else {
      this.currentContext = null;
    }
  }
  
  /**
   * 获取当前上下文
   */
  getCurrentContext(): TraceContext | null {
    return this.currentContext;
  }
  
  /**
   * 注入到 HTTP Headers
   */
  injectIntoHeaders(headers: Record<string, string>): void {
    const context = this.currentContext;
    if (!context) return;
    
    headers['X-Trace-ID'] = context.traceId;
    headers['X-Span-ID'] = context.spanId;
    headers['X-Sampled'] = context.sampled ? '1' : '0';
  }
  
  /**
   * 从 HTTP Headers 提取
   */
  extractFromHeaders(headers: Record<string, string>): TraceContext | null {
    const traceId = headers['X-Trace-ID'];
    if (!traceId) return null;
    
    return {
      traceId,
      spanId: headers['X-Span-ID'] || this.generateSpanId(),
      parentSpanId: headers['X-Parent-Span-ID'],
      sampled: headers['X-Sampled'] === '1',
    };
  }
}
```

### 3.2 环形缓冲区 (Ring Buffer)

```typescript
/**
 * 高性能环形缓冲区
 * 
 * 特点：
 * 1. 无锁写入 (仅使用原子操作)
 * 2. 批量消费
 * 3. 固定内存占用
 */
class RingBuffer<T> {
  private buffer: T[];
  private capacity: number;
  private head: number = 0;  // 写入位置
  private tail: number = 0;  // 读取位置
  private count: number = 0;  // 当前元素数量
  
  constructor(capacity: number) {
    this.capacity = capacity;
    this.buffer = new Array(capacity);
  }
  
  /**
   * 写入单个元素
   * O(1) 时间复杂度
   */
  push(item: T): boolean {
    if (this.count >= this.capacity) {
      return false;  // 缓冲区满
    }
    
    this.buffer[this.head] = item;
    this.head = (this.head + 1) % this.capacity;
    this.count++;
    
    return true;
  }
  
  /**
   * 批量写入
   */
  pushBatch(items: T[]): number {
    let written = 0;
    for (const item of items) {
      if (this.push(item)) {
        written++;
      } else {
        break;
      }
    }
    return written;
  }
  
  /**
   * 读取单个元素
   */
  pop(): T | undefined {
    if (this.count === 0) {
      return undefined;
    }
    
    const item = this.buffer[this.tail];
    this.buffer[this.tail] = undefined as any;  // 释放引用
    this.tail = (this.tail + 1) % this.capacity;
    this.count--;
    
    return item;
  }
  
  /**
   * 批量读取
   */
  popBatch(maxCount: number): T[] {
    const items: T[] = [];
    const count = Math.min(maxCount, this.count);
    
    for (let i = 0; i < count; i++) {
      const item = this.pop();
      if (item !== undefined) {
        items.push(item);
      }
    }
    
    return items;
  }
  
  /**
   * 查看但不移除
   */
  peek(): T | undefined {
    if (this.count === 0) {
      return undefined;
    }
    return this.buffer[this.tail];
  }
  
  get size(): number {
    return this.count;
  }
  
  get isEmpty(): boolean {
    return this.count === 0;
  }
  
  get isFull(): boolean {
    return this.count >= this.capacity;
  }
}
```

### 3.3 智能采样器

```typescript
/**
 * 采样策略
 */
interface SamplingStrategy {
  /**
   * 是否采样
   */
  shouldSample(record: LogRecord): boolean;
  
  /**
   * 获取采样率
   */
  getSampleRate(): number;
}

/**
 * 采样决策结果
 */
interface SamplingDecision {
  sampled: boolean;
  reason: string;
  sampleRate: number;
}

/**
 * 组合采样器
 * 
 * 规则优先级：
 * 1. 错误日志 100% 保留
 * 2. 特定模块 100% 保留
 * 3. 默认采样率
 */
class CompositeSampler implements SamplingStrategy {
  private rules: Array<{
    matcher: (record: LogRecord) => boolean;
    sampleRate: number;
  }> = [];
  
  private defaultRate: number = 0.1;  // 默认 10%
  
  /**
   * 添加采样规则
   */
  addRule(
    matcher: (record: LogRecord) => boolean,
    sampleRate: number
  ): this {
    this.rules.push({ matcher, sampleRate });
    return this;
  }
  
  /**
   * 设置默认采样率
   */
  setDefaultRate(rate: number): this {
    this.defaultRate = rate;
    return this;
  }
  
  /**
   * 判断是否采样
   */
  shouldSample(record: LogRecord): boolean {
    // 1. 错误日志总是采样
    if (record.level >= LogLevel.ERROR) {
      return true;
    }
    
    // 2. 致命错误总是采样
    if (record.level === LogLevel.FATAL) {
      return true;
    }
    
    // 3. 应用规则
    for (const rule of this.rules) {
      if (rule.matcher(record)) {
        return Math.random() < rule.sampleRate;
      }
    }
    
    // 4. 默认采样率
    return Math.random() < this.defaultRate;
  }
  
  getSampleRate(): number {
    return this.defaultRate;
  }
}

/**
 * 速率限制采样器
 * 
 * 限制每秒最大日志数
 */
class RateLimitSampler implements SamplingStrategy {
  private maxPerSecond: number;
  private tokens: number;
  private lastRefill: number;
  
  constructor(maxPerSecond: number = 1000) {
    this.maxPerSecond = maxPerSecond;
    this.tokens = maxPerSecond;
    this.lastRefill = Date.now();
    
    // 每秒补充令牌
    setInterval(() => {
      this.tokens = this.maxPerSecond;
      this.lastRefill = Date.now();
    }, 1000);
  }
  
  shouldSample(): boolean {
    if (this.tokens >= 1) {
      this.tokens--;
      return true;
    }
    return false;
  }
  
  getSampleRate(): number {
    return this.tokens / this.maxPerSecond;
  }
}

// 预定义采样器
const DefaultSampler = new CompositeSampler()
  .addRule(record => record.context?.module === 'security', 1.0)  // 安全日志 100%
  .addRule(record => record.context?.module === 'auth', 1.0)    // 认证日志 100%
  .addRule(record => record.context?.module === 'payment', 1.0)  // 支付日志 100%
  .setDefaultRate(0.1);  // 默认 10%
```

### 3.4 异步 Transport

```typescript
/**
 * 异步 Remote Transport
 * 
 * 特性：
 * 1. 批量发送
 * 2. 自动重试
 * 3. 背压处理
 */
class AsyncRemoteTransport implements ITransport {
  private readonly config: RemoteTransportConfig;
  private readonly formatter: IFormatter;
  private readonly buffer: RingBuffer<string>;
  private readonly httpClient: HttpClient;
  private readonly retryQueue: Array<{ record: LogRecord; retries: number }> = [];
  
  private flushTimer: ReturnType<typeof setInterval> | null = null;
  private isFlushing: boolean = false;
  private backpressure: boolean = false;
  
  constructor(config: RemoteTransportConfig, formatter: IFormatter) {
    this.config = config;
    this.formatter = formatter;
    this.buffer = new RingBuffer(10000);
    
    this.httpClient = new HttpClient({
      baseURL: config.url,
      headers: config.headers,
      timeout: 5000,
    });
    
    // 启动定时刷新
    this.startFlushTimer();
  }
  
  /**
   * 写入日志
   */
  async write(record: LogRecord): Promise<void> {
    const formatted = this.formatter.format(record);
    
    // 背压处理
    if (this.backpressure && record.level < LogLevel.ERROR) {
      return;  // 丢弃非错误日志
    }
    
    // 尝试写入缓冲区
    if (!this.buffer.push(formatted)) {
      // 缓冲区满，触发背压
      this.backpressure = true;
      
      // 等待缓冲区有空位
      await this.waitForSpace();
    }
  }
  
  /**
   * 批量刷新
   */
  private async flush(): Promise<void> {
    if (this.isFlushing) return;
    this.isFlushing = true;
    
    try {
      // 获取批量数据
      const records = this.buffer.popBatch(this.config.batch?.size || 100);
      
      if (records.length === 0) {
        return;
      }
      
      // 发送请求
      await this.sendBatch(records);
      
      // 背压解除
      if (this.buffer.size < this.buffer.capacity * 0.5) {
        this.backpressure = false;
      }
      
    } catch (error) {
      // 重试处理
      await this.handleFlushError(error);
    } finally {
      this.isFlushing = false;
    }
  }
  
  /**
   * 发送批量日志
   */
  private async sendBatch(records: string[]): Promise<void> {
    const maxRetries = this.config.retry?.maxRetries || 3;
    
    for (let attempt = 0; attempt < maxRetries; attempt++) {
      try {
        await this.httpClient.post('/logs/batch', {
          logs: records,
          timestamp: Date.now(),
        });
        
        return;  // 成功
        
      } catch (error) {
        if (attempt === maxRetries - 1) {
          // 最后的重试也失败，记录到重试队列
          for (const record of records) {
            this.retryQueue.push({ record: JSON.parse(record), retries: maxRetries });
          }
        }
        
        // 指数退避
        const delay = Math.pow(2, attempt) * 1000;
        await sleep(delay);
      }
    }
  }
  
  /**
   * 处理刷新错误
   */
  private async handleFlushError(error: unknown): Promise<void> {
    // 错误日志不能丢失
    const errorRecords = this.buffer.popBatch(100).map(r => JSON.parse(r));
    
    // 重试队列中的记录
    const retryFailed: typeof this.retryQueue = [];
    
    for (const item of this.retryQueue) {
      if (item.retries > 0) {
        item.retries--;
        retryFailed.push(item);
      } else {
        // 重试次数用完，记录到本地文件
        this.writeToLocalFile(item.record);
      }
    }
    
    this.retryQueue.length = 0;
    this.retryQueue.push(...retryFailed);
  }
  
  /**
   * 等待缓冲区空间
   */
  private async waitForSpace(): Promise<void> {
    while (this.buffer.isFull) {
      await sleep(100);
    }
  }
  
  /**
   * 启动定时刷新
   */
  private startFlushTimer(): void {
    const interval = this.config.batch?.flushInterval || 1000;
    
    this.flushTimer = setInterval(() => {
      this.flush().catch(console.error);
    }, interval);
    
    this.flushTimer.unref();
  }
}
```

```typescript
/**
 * Elasticsearch Transport

class ElasticsearchTransport implements ITransport {
  private readonly client: Client;
  private readonly indexPattern: string;
  private readonly buffer: Buffer;
  
  constructor(config: {
    node: string;
    indexPattern: string;
    username?: string;
    password?: string;
    batchSize?: number;
  }, formatter: IFormatter) {
    this.client = new Client({
      node: config.node,
      auth: config.username && config.password
        ? { username: config.username, password: config.password }
        : undefined,
    });
    
    this.indexPattern = config.indexPattern || 'agent-logs-%{YYYY.MM.DD}';
    this.buffer = new Buffer(config.batchSize || 100);
  }
  
  async write(record: LogRecord): Promise<void> {
    const index = this.resolveIndex(this.indexPattern);
    const document = {
      '@timestamp': record.timestamp,
      level: record.levelName,
      message: record.message,
      ...record.context,
      trace: record.trace,
      request: record.request,
      agent: record.agent,
      error: record.error,
    };
    
    this.buffer.add({ index, document });
    
    if (this.buffer.isFull) {
      await this.flush();
    }
  }
  
  private resolveIndex(pattern: string): string {
    const now = new Date();
    return pattern.replace('%{YYYY}', now.getFullYear().toString())
      .replace('%{MM}', (now.getMonth() + 1).toString().padStart(2, '0'))
      .replace('%{DD}', now.getDate().toString().padStart(2, '0'));
  }
  
  async flush(): Promise<void> {
    if (this.buffer.isEmpty) return;
    
    const body = this.buffer.getAll().flatMap(doc => [
      { index: { _index: doc.index } },
      doc.document
    ]);
    
    await this.client.bulk({ refresh: true });
    this.buffer.clear();
  }
}
```

```typescript
/**
 * Loki Transport
 */
class LokiTransport implements ITransport {
  private readonly client: HttpClient;
  private readonly url: string;
  private readonly buffer: Array<{ labels: string; content: string }> = [];
  private readonly maxBufferSize: number = 1000;
  
  constructor(config: {
    url: string;
    username?: string;
    password?: string;
    labels?: Record<string, string>;
  }, formatter: IFormatter) {
    this.url = config.url;
    this.client = new HttpClient({
      headers: {
        'Content-Type': 'application/json',
        ...(config.username && {
          'Authorization': `Basic ${Buffer.from(`${config.username}:${config.password}`).toString('base64')}`
        })
      }
    });
    
    this.defaultLabels = config.labels || {};
  }
  
  async write(record: LogRecord): Promise<void> {
    const labels = {
      ...this.defaultLabels,
      level: record.levelName,
      service: record.context?.service,
    };
    
    const stream = JSON.stringify(labels);
    const content = this.formatter.format(record);
    
    this.buffer.push({ labels: stream, content });
    
    if (this.buffer.length >= this.maxBufferSize) {
      await this.flush();
    }
  }
  
  async flush(): Promise<void> {
    if (this.buffer.length === 0) return;
    
    const streams: Record<string, string[]> = {};
    
    for (const { labels, content } of this.buffer) {
      if (!streams[labels]) {
        streams[labels] = [];
      }
      streams[labels].push(content);
    }
    
    const payload = {
      streams: Object.entries(streams).map(([labels, values]) => ({
        stream: JSON.parse(labels),
        values: values.map((v, i) => [
          Date.now() * 1000000 + i,  // 纳秒时间戳
          v
        ])
      }))
    };
    
    await this.client.post('/loki/api/v1/push', payload);
    this.buffer.length = 0;
  }
}
```

```typescript
/**
 * 日志聚合器
 * 
 * 实时统计日志指标
 */
class LogAggregator {
  private counters = new Map<string, number>();
  private timers = new Map<string, number[]>();
  private errorCounts = new Map<string, number>();
  
  /**
   * 记录指标
   */
  record(record: LogRecord): void {
    // 计数器
    const counterKey = `logs.${record.levelName.toLowerCase()}`;
    this.counters.set(counterKey, (this.counters.get(counterKey) || 0) + 1);
    
    // 模块计数器
    const module = record.context?.module;
    if (module) {
      const moduleKey = `logs.module.${module}`;
      this.counters.set(moduleKey, (this.counters.get(moduleKey) || 0) + 1);
    }
    
    // 延迟统计
    if (record.request?.duration) {
      const durationKey = 'logs.duration';
      const durations = this.timers.get(durationKey) || [];
      durations.push(record.request.duration);
      
      // 保留最近 1000 个样本
      if (durations.length > 1000) {
        durations.shift();
      }
      this.timers.set(durationKey, durations);
    }
    
    // 错误统计
    if (record.level >= LogLevel.ERROR) {
      const errorKey = record.error?.name || 'UnknownError';
      this.errorCounts.set(errorKey, (this.errorCounts.get(errorKey) || 0) + 1);
    }
  }
  
  /**
   * 获取统计指标
   */
  getMetrics(): AggregatedMetrics {
    return {
      counters: Object.fromEntries(this.counters),
      timers: Object.fromEntries(
        Array.from(this.timers.entries()).map(([key, values]) => [
          key,
          {
            count: values.length,
            sum: values.reduce((a, b) => a + b, 0),
            avg: values.reduce((a, b) => a + b, 0) / values.length,
            p50: this.percentile(values, 0.5),
            p95: this.percentile(values, 0.95),
            p99: this.percentile(values, 0.99),
          }
        ])
      ),
      errors: Object.fromEntries(this.errorCounts),
    };
  }
  
  private percentile(values: number[], p: number): number {
    const sorted = [...values].sort((a, b) => a - b);
    const index = Math.ceil(sorted.length * p) - 1;
    return sorted[Math.max(0, index)];
  }
  
  /**
   * 重置统计
   */
  reset(): void {
    this.counters.clear();
    this.timers.clear();
    this.errorCounts.clear();
  }
}
```

### 6.2 告警规则

```typescript
/**
 * 告警规则
 */
interface AlertRule {
  name: string;
  condition: (metrics: AggregatedMetrics) => boolean;
  severity: 'warning' | 'critical';
  message: string;
  cooldown: number;  // 冷却时间 (毫秒)
}

/**
 * 告警管理器
 */
class AlertManager {
  private rules: AlertRule[] = [];
  private lastAlertTime = new Map<string, number>();
  private notifier: AlertNotifier;
  
  /**
   * 注册告警规则
   */
  registerRule(rule: AlertRule): void {
    this.rules.push(rule);
  }
  
  /**
   * 检查告警
   */
  async checkAlerts(metrics: AggregatedMetrics): Promise<void> {
    for (const rule of this.rules) {
      try {
        if (rule.condition(metrics)) {
          await this.triggerAlert(rule);
        }
      } catch (error) {
        console.error(`Alert check failed for ${rule.name}:`, error);
      }
    }
  }
  
  /**
   * 触发告警
   */
  private async triggerAlert(rule: AlertRule): Promise<void> {
    // 检查冷却时间
    const lastTime = this.lastAlertTime.get(rule.name) || 0;
    if (Date.now() - lastTime < rule.cooldown) {
      return;
    }
    
    this.lastAlertTime.set(rule.name, Date.now());
    
    // 发送告警
    await this.notifier.notify({
      name: rule.name,
      severity: rule.severity,
      message: rule.message,
      timestamp: Date.now(),
      metrics: this.getSnapshot(),
    });
  }
}

// 预定义告警规则
const DefaultAlertRules: AlertRule[] = [
  {
    name: 'high_error_rate',
    severity: 'critical',
    cooldown: 60000,
    condition: (metrics) => {
      const total = metrics.counters['logs.info'] || 0;
      const errors = metrics.counters['logs.error'] || 0;
      return total > 100 && (errors / total) > 0.05;  // 错误率 > 5%
    },
    message: '错误率超过 5%',
  },

    name: 'high_latency',
    severity: 'warning',
    cooldown: 120000,
    condition: (metrics) => {
      const p99 = metrics.timers['logs.duration']?.p99;
      return p99 !== undefined && p99 > 5000;  // P99 延迟 > 5s
    },
    message: 'P99 延迟超过 5 秒',
  },
  {
    name: 'too_many_errors',
    severity: 'critical',
    cooldown: 300000,
    condition: (metrics) => {
      return (metrics.counters['logs.error'] || 0) > 100;
    },
    message: '错误数量过多',
```

### 8.2 适配现有 Logger

```typescript
/**
 * 适配器：将现有 Logger 适配到新的高性能日志系统
 */
class HighPerformanceLoggerAdapter {
  private aggregator: LogAggregator;
  private alertManager: AlertManager;
  private transports: ITransport[];
  
  constructor(config: LoggerConfig) {
    this.aggregator = new LogAggregator();
    this.alertManager = new AlertManager(config.alerts);
    
    // 初始化 Transport
    this.transports = this.createTransports(config);
    
    // 启动告警检查定时器
    this.startAlertChecker();
  }
  
  /**
   * 处理日志记录
   */
  handleLog(record: LogRecord): void {
    // 1. 聚合统计
    this.aggregator.record(record);
    
    // 2. 写入各 Transport
    for (const transport of this.transports) {
      transport.write(record).catch(console.error);
    }
  }
  
  /**
   * 创建 Transport 列表
   */
  private createTransports(config: LoggerConfig): ITransport[] {
    const transports: ITransport[] = [];
    const formatter = new JsonFormatter();
    
    if (config.console?.enabled) {
      transports.push(new ConsoleTransport(config.console, new PrettyFormatter()));
    }
    
    if (config.file?.enabled) {
      transports.push(new FileTransport(config.file, formatter));
    }
    
    if (config.remote?.enabled) {
      transports.push(new AsyncRemoteTransport(config.remote, formatter));
    }
    
    return transports;
  }
  
  /**
   * 启动告警检查
   */
  private startAlertChecker(): void {
    setInterval(() => {
      const metrics = this.aggregator.getMetrics();
      this.alertManager.checkAlerts(metrics);
    }, 10000);  // 每 10 秒检查
  }
}
```
