## 1. 模块概述

### 1.1 设计目标

认证授权模块是企业级 Agent 服务端的安全核心，负责：

| 目标 | 描述 | 优先级 |
|------|------|--------|
| **身份认证** | 验证请求者身份 | P0 |
| **权限控制** | 基于角色的访问控制 (RBAC) | P0 |
| **多租户隔离** | 租户间数据完全隔离 | P0 |
| **审计日志** | 记录所有敏感操作 | P1 |
| **安全合规** | 符合企业安全要求 | P0 |

### 1.2 核心组件

```
认证授权模块
│
├── Auth Service (认证服务)
│   ├── API Key 认证
│   ├── JWT 认证
│   └── OAuth 2.0 (可选)
│
├── Permission Service (权限服务)
│   ├── 角色管理
│   ├── 权限检查

├── Tenant Service (租户服务)
│   ├── 租户管理

│
└── Rate Limit Service (限流服务)
    ├── 请求限流
    ├── Token 限流
    └── 并发限制
```

### 3.2 认证中间件实现

```typescript
/**
 * 认证中间件
 * 
 * 支持多种认证方式：
 * 1. API Key (X-API-Key header)
 * 2. Bearer Token (Authorization header)
 * 3. Cookie (refresh_token)
 */
class AuthMiddleware {
  private apiKeyService: ApiKeyService;
  private jwtService: JwtService;
  private logger: Logger;
  
  /**
   * 认证处理
   */
  async authenticate(request: FastifyRequest): Promise<AuthContext> {
    // 1. 尝试 API Key
    const apiKey = request.headers['x-api-key'];
    if (apiKey) {
      return this.authenticateByApiKey(apiKey as string);
    }
    
    // 2. 尝试 Bearer Token
    const authHeader = request.headers.authorization;
    if (authHeader?.startsWith('Bearer ')) {
      return this.authenticateByJwt(authHeader.slice(7));
    }
    
    // 3. 尝试 Cookie
    const cookieToken = request.cookies?.refresh_token;
    if (cookieToken) {
      return this.authenticateByRefreshToken(cookieToken);
    }
    
    // 无认证信息
    throw new UnauthorizedError('No authentication credentials provided');
  }
  
  /**
   * API Key 认证
   */
  private async authenticateByApiKey(key: string): Promise<AuthContext> {
    // 1. 哈希 API Key
    const keyHash = this.hashApiKey(key);
    
    // 2. 查找 API Key
    const apiKey = await this.apiKeyService.findByHash(keyHash);
    if (!apiKey) {
      throw new UnauthorizedError('Invalid API Key');
    }
    
    // 3. 检查状态
    if (apiKey.status === 'revoked') {
      throw new UnauthorizedError('API Key has been revoked');
    }
    
    // 4. 检查过期
    if (apiKey.expiresAt && new Date(apiKey.expiresAt) < new Date()) {
      throw new UnauthorizedError('API Key has expired');
    }
    
    // 5. 更新使用记录
    await this.apiKeyService.recordUsage(apiKey.id);
    
    // 6. 构建上下文
    return {
      type: 'api_key',
      tenantId: apiKey.tenantId,
      userId: apiKey.userId,
      role: 'user',
      permissions: apiKey.permissions,
      apiKeyId: apiKey.id,
    };
  }
  
  /**
   * JWT 认证
   */
  private async authenticateByJwt(token: string): Promise<AuthContext> {
    try {
      // 1. 验证 Token
      const payload = await this.jwtService.verifyAccessToken(token);
      
      // 2. 检查过期
      if (payload.exp * 1000 < Date.now()) {
        throw new UnauthorizedError('Token has expired');
      }
      
      // 3. 返回上下文
      return {
        type: 'jwt',
        tenantId: payload.tenantId,
        userId: payload.sub,
        role: payload.role,
        permissions: payload.permissions,
        tokenId: payload.jti,
      };
    } catch (error) {
      if (error instanceof UnauthorizedError) {
        throw error;
      }
      throw new UnauthorizedError('Invalid token');
    }
  }
  
  /**
   * Refresh Token 认证
   */
  private async authenticateByRefreshToken(token: string): Promise<AuthContext> {
    try {
      // 1. 验证 Refresh Token
      const payload = await this.jwtService.verifyRefreshToken(token);
      
      // 2. 检查是否需要轮换
      if (this.refreshTokenConfig.rotateOnUse) {
        // 生成新的 Refresh Token
        await this.jwtService.rotateRefreshToken(payload.jti);
      }
      
      // 3. 返回上下文
      return {
        type: 'refresh',
        tenantId: payload.tenantId,
        userId: payload.sub,
        role: payload.role,
        permissions: payload.permissions,
        tokenId: payload.jti,
      };
    } catch (error) {
      throw new UnauthorizedError('Invalid refresh token');
    }
  }
  
  /**
   * 哈希 API Key
   */
  private hashApiKey(key: string): string {
    return crypto
      .createHash('sha256')
      .update(key + this.config.apiKeySalt)
      .digest('hex');
  }
}

// 认证上下文
interface AuthContext {
  type: 'api_key' | 'jwt' | 'refresh';
  tenantId: string;
  userId: string;
  role: UserRole;
  permissions: Permission[];
  apiKeyId?: string;
  tokenId?: string;
}
```

### 3.3 JWT 服务实现

```typescript
/**
 * JWT 服务
 */
class JwtService {
  private redis: Redis;
  private config: JwtConfig;
  
  // Access Token 密钥
  private readonly ACCESS_TOKEN_EXPIRY = 15 * 60;        // 15 分钟
  private readonly REFRESH_TOKEN_EXPIRY = 7 * 24 * 3600; // 7 天
  
  /**
   * 生成 Access Token
   */
  async generateAccessToken(params: {
    userId: string;
    tenantId: string;
    role: UserRole;
    permissions: Permission[];
  }): Promise<string> {
    const payload: JwtPayload = {
      sub: params.userId,
      iss: this.config.issuer,
      aud: params.tenantId,
      exp: Math.floor(Date.now() / 1000) + this.ACCESS_TOKEN_EXPIRY,
      iat: Math.floor(Date.now() / 1000),
      jti: generateUUID(),
      tenantId: params.tenantId,
      userId: params.userId,
      role: params.role,
      permissions: params.permissions,
      type: 'access',
    };
    
    // 添加到黑名单 (用于注销)
    await this.redis.setex(
      `jwt:blacklist:${payload.jti}`,
      this.ACCESS_TOKEN_EXPIRY,
      '1'
    );
    
    return this.sign(payload);
  }
  
  /**
   * 生成 Refresh Token
   */
  async generateRefreshToken(params: {
    userId: string;
    tenantId: string;
    role: UserRole;
    permissions: Permission[];
  }): Promise<string> {
    const payload: JwtPayload = {
      sub: params.userId,
      iss: this.config.issuer,
      aud: params.tenantId,
      exp: Math.floor(Date.now() / 1000) + this.REFRESH_TOKEN_EXPIRY,
      iat: Math.floor(Date.now() / 1000),
      jti: generateUUID(),
      tenantId: params.tenantId,
      userId: params.userId,
      role: params.role,
      permissions: params.permissions,
      type: 'refresh',
    };
    
    // 存储 Refresh Token
    await this.redis.setex(
      `jwt:refresh:${payload.jti}`,
      this.REFRESH_TOKEN_EXPIRY,
      JSON.stringify(payload)
    );
    
    return this.sign(payload);
  }
  
  /**
   * 验证 Access Token
   */
  async verifyAccessToken(token: string): Promise<JwtPayload> {
    // 1. 解码 Token
    const decoded = this.decode(token);
    
    // 2. 检查是否在黑名单
    const isBlacklisted = await this.redis.get(`jwt:blacklist:${decoded.jti}`);
    if (isBlacklisted) {
      throw new UnauthorizedError('Token has been revoked');
    }
    
    // 3. 验证签名
    const payload = await this.verify(token, decoded);
    
    // 4. 检查类型
    if (payload.type !== 'access') {
      throw new UnauthorizedError('Invalid token type');
    }
    
    return payload;
  }
  
  /**
   * 验证 Refresh Token
   */
  async verifyRefreshToken(token: string): Promise<JwtPayload> {
    const decoded = this.decode(token);
    
    // 检查是否存储
    const stored = await this.redis.get(`jwt:refresh:${decoded.jti}`);
    if (!stored) {
      throw new UnauthorizedError('Refresh token has been used or expired');
    }
    
    // 验证签名
    const payload = await this.verify(token, decoded);
    
    if (payload.type !== 'refresh') {
      throw new UnauthorizedError('Invalid token type');
    }
    
    return payload;
  }
  
  /**
   * 轮换 Refresh Token
   */
  async rotateRefreshToken(tokenId: string): Promise<void> {
    // 删除旧的 Refresh Token
    await this.redis.del(`jwt:refresh:${tokenId}`);
  }
  
  /**
   * 注销 Token (加入黑名单)
   */
  async revokeToken(tokenId: string, type: 'access' | 'refresh'): Promise<void> {
    if (type === 'access') {
      // Access Token 直接加入黑名单
      await this.redis.setex(
        `jwt:blacklist:${tokenId}`,
        this.ACCESS_TOKEN_EXPIRY,
        '1'
      );
    } else {
      // Refresh Token 直接删除
      await this.redis.del(`jwt:refresh:${tokenId}`);
    }
  }
  
  /**
   * 签名
   */
  private sign(payload: JwtPayload): string {
    return jwt.sign(payload, this.config.secret, {
      algorithm: 'HS256',
    });
  }
  
  /**
   * 验证
   */
  private async verify(token: string, decoded: any): Promise<JwtPayload> {
    try {
      return jwt.verify(token, this.config.secret, {
        algorithms: ['HS256'],
      }) as JwtPayload;
    } catch (error) {
      throw new UnauthorizedError('Invalid token signature');
    }
  }
  
  /**
   * 解码
   */
  private decode(token: string): any {
    return jwt.decode(token);
  }
}
```

```typescript
/**
 * 权限服务
 */
class PermissionService {
  /**
   * 检查权限
   */
  async checkPermission(
    authContext: AuthContext,
    permission: Permission,
    resource?: Resource
  ): Promise<boolean> {
    // 1.用户权限
    if (authContext.permissions.includes(permission)) {
 检查      return true;
    }
    
    // 2. 检查特殊权限 (如读取所有会话)
    if (permission === Permission.SESSION_READ_ALL && resource) {
      return this.checkSessionReadAll(authContext, resource);
    }
    
    return false;
  }
  
  /**
   * 检查会话读取权限 (跨租户/跨用户)
   */
  private async checkSessionReadAll(
    authContext: AuthContext,
    resource: Resource
  ): Promise<boolean> {
    // Owner 和 Admin 可以读取所有会话
    if (['owner', 'admin'].includes(authContext.role)) {
      return true;
    }
    
    // Developer 只能读取自己的会话
    if (authContext.role === 'developer') {
      return resource.userId === authContext.userId;
    }
    
    // User 只能读取自己的会话
    return resource.sessionTenantId === authContext.tenantId &&
           resource.userId === authContext.userId;
  }
  
  /**
   * 批量检查权限
   */
  async checkPermissions(
    authContext: AuthContext,
    permissions: Permission[]
  ): Promise<Record<Permission, boolean>> {
    const results: Record<Permission, boolean> = {} as any;
    
    for (const perm of permissions) {
      results[perm] = await this.checkPermission(authContext, perm);
    }
    
    return results;
  }
}

      );
      
      if (!hasPermission) {
        throw new ForbiddenError(`Missing permission: ${permission}`);
      }
      
      return originalMethod.apply(this, args);
    };
    
    return descriptor;
  };
}
```

```typescript
class TenantService {
  private db: PrismaClient;
  private quotaService: QuotaService;

   * 验证租户状态
   */
  async validateTenant(tenantId: string): Promise<Tenant> {

    });

      throw new NotFoundError('Tenant not found');
    }

      throw new TenantSuspendedError('Tenant is suspended');
    }
    
    if (tenant.status === 'deleted') {
      throw new NotFoundError('Tenant not found');
    }
    
    if (tenant.status === 'pending') {
      throw new TenantPendingError('Tenant is not activated');
    }
    
    return tenant;
  }
  
  /**
   * 获取租户配额
   */
  async getTenantQuotas(tenantId: string): Promise<TenantQuotas> {
    const tenant = await this.validateTenant(tenantId);
    
    // 使用套餐默认配额 + 自定义配额
    const planQuotas = PLAN_QUOTAS[tenant.plan];
    
    return {
      ...planQuotas,
      ...tenant.customQuotas,
    };
  }
  
  /**
   * 检查配额
   */
  async checkQuota(
    tenantId: string,
    quotaType: keyof TenantQuotas,
    value: number = 1
  ): Promise<void> {
    const quotas = await this.getTenantQuotas(tenantId);
    const limit = quotas[quotaType];
    
    // -1 表示无限制
    if (limit === -1) {
      return;
    }
    
    // 获取当前使用量
    const usage = await this.quotaService.getUsage(tenantId, quotaType);
    
    if (usage + value > limit) {
      throw new QuotaExceededError(
        `Quota exceeded: ${quotaType}`,
        {
          type: quotaType,
          limit,
          used: usage,
          requested: value,
        }
      );
    }
  }
  
  /**
   * 记录配额使用
   */
  async recordUsage(
    tenantId: string,
    quotaType: keyof TenantQuotas,
    value: number
  ): Promise<void> {
    await this.quotaService.record(tenantId, quotaType, value);
  }
}

/**
 * 配额服务
 */
class QuotaService {
  private redis: Redis;
  private db: PrismaClient;
  
  /**
   * 获取当前使用量
   */
  async getUsage(
    tenantId: string,
    quotaType: keyof TenantQuotas
  ): Promise<number> {
    const today = this.getDateKey();
    const key = `quota:${tenantId}:${quotaType}:${today}`;
    
    // 优先从 Redis 获取
    const cached = await this.redis.get(key);
    if (cached) {
      return parseInt(cached, 10);
    }
    
    // 从数据库聚合
    const result = await this.db.usageRecord.aggregate({
      where: {
        tenantId,
        quotaType,
        date: new Date(today),
      },
      _sum: {
        value: true,
      },
    });
    
    const usage = result._sum.value || 0;
    
    // 缓存到 Redis
    await this.redis.setex(key, 86400, usage.toString());
    
    return usage;
  }
  
  /**
   * 记录使用量
   */
  async record(
    tenantId: string,
    quotaType: keyof TenantQuotas,
    value: number
  ): Promise<void> {
    const today = this.getDateKey();
    
    // 1. 更新 Redis 计数器
    const redisKey = `quota:${tenantId}:${quotaType}:${today}`;
    await this.redis.incrby(redisKey, value);
    await this.redis.expire(redisKey, 86400 * 2); // 保留 2 天
    
    // 2. 记录到数据库 (异步)
    this.db.usageRecord.create({
      data: {
        tenantId,
        quotaType,
        value,
        date: new Date(today),
      },
    }).catch(err => {
      // 记录错误但不影响主流程
      console.error('Failed to record quota:', err);
    });
  }
  
  /**
   * 获取日期键
   */
  private getDateKey(): string {
    return new Date().toISOString().split('T')[0];
  }
}
```

### 6.1 限流策略

```typescript
/**
 * 限流服务
 */
class RateLimitService {
  private redis: Redis;
  
  /**
   * 检查请求限流
   * 
   * 使用滑动窗口算法
   */
  async checkRateLimit(
    tenantId: string,
    userId: string,
    options: {
      windowMs: number;        // 窗口大小 (毫秒)
      maxRequests: number;    // 最大请求数
    }
  ): Promise<RateLimitResult> {
    const { windowMs, maxRequests } = options;
    
    // 无限制
    if (maxRequests === -1) {
      return { allowed: true, remaining: -1, resetAt: 0 };
    }
    
    const key = `ratelimit:${tenantId}:${userId}`;
    const now = Date.now();
    const windowStart = now - windowMs;
    
    // 1. 移除过期的记录
    await this.redis.zRemRangeByScore(key, 0, windowStart.toString());
    
    // 2. 获取当前请求数
    const count = await this.redis.zCard(key);
    
    // 3. 检查是否超限
    if (count >= maxRequests) {
      // 获取最旧的请求时间
      const oldest = await this.redis.zRange(key, 0, 0);
      const resetAt = oldest.length > 0 
        ? Number(oldest[0]) + windowMs 
        : now + windowMs;
      
      return {
        allowed: false,
        remaining: 0,
        resetAt,
        retryAfter: Math.ceil((resetAt - now) / 1000),
      };
    }
    
    // 4. 添加当前请求
    await this.redis.zAdd(key, {
      score: now,
      value: now.toString(),
    });
    await this.redis.expire(key, Math.ceil(windowMs / 1000) + 1);
    
    return {
      allowed: true,
      remaining: maxRequests - count - 1,
      resetAt: now + windowMs,
      retryAfter: 0,
    };
  }
  
  /**
   * 检查 Token 限流
   */
  async checkTokenLimit(
    tenantId: string,
    tokens: number
  ): Promise<RateLimitResult> {
    const today = this.getDateKey();
    const key = `ratelimit:tokens:${tenantId}:${today}`;
    
    // 获取每日限额
    const dailyLimit = await this.getDailyTokenLimit(tenantId);
    
    if (dailyLimit === -1) {
      return { allowed: true, remaining: -1, resetAt: 0 };
    }
    
    // 获取已使用
    const used = parseInt(await this.redis.get(key) || '0', 10);
    
    if (used + tokens > dailyLimit) {
      const resetAt = new Date().setHours(23, 59, 59, 999);
      return {
        allowed: false,
        remaining: Math.max(0, dailyLimit - used),
        resetAt,
        retryAfter: Math.ceil((resetAt - Date.now()) / 1000),
      };
    }
    
    // 消耗 Token
    await this.redis.incrby(key, tokens);
    await this.redis.expire(key, 86400 * 2);
    
    return {
      allowed: true,
      remaining: dailyLimit - used - tokens,
      resetAt: new Date().setHours(23, 59, 59, 999),
      retryAfter: 0,
    };
  }
  
  /**
   * 并发限制
   */
  async checkConcurrentLimit(
    tenantId: string
  ): Promise<{ allowed: boolean; current: number; limit: number }> {
    const key = `ratelimit:concurrent:${tenantId}`;
    
    const limit = await this.getConcurrentLimit(tenantId);
    
    if (limit === -1) {
      return { allowed: true, current: 0, limit: -1 };
    }
    
    const current = parseInt(await this.redis.get(key) || '0', 10);
    
    if (current >= limit) {
      return { allowed: false, current, limit };
    }
    
    // 增加计数
    await this.redis.incr(key);
    
    return { allowed: true, current: current + 1, limit };
  }
  
  /**
   * 释放并发限制
   */
  async releaseConcurrentLimit(tenantId: string): Promise<void> {
    const key = `ratelimit:concurrent:${tenantId}`;
    await this.redis.decr(key);
  }
}

// 限流结果
interface RateLimitResult {
  allowed: boolean;
  remaining: number;
  resetAt: number;       // 毫秒时间戳
  retryAfter?: number;   // 秒
}
```

```typescript
/**
 * API Key 服务
 */
class ApiKeyService {
  private db: PrismaClient;
  private redis: Redis;
  private logger: Logger;
  
  private readonly KEY_PREFIX = 'sk_';    // 显示前缀
  private readonly KEY_LENGTH = 32;       // Key 长度
  
  /**
   * 创建 API Key
   */
  async create(params: {
    tenantId: string;
    userId: string;
    name: string;
    permissions: Permission[];
    expiresIn?: number;       // 天数
  }): Promise<CreateApiKeyResponse> {
    // 1. 生成随机 Key
    const key = this.generateKey();
    
    // 2. 哈希存储
    const keyHash = this.hashKey(key);
    
    // 3. 保存到数据库
    const apiKey = await this.db.apiKey.create({
      data: {
        tenantId: params.tenantId,
        userId: params.userId,
        name: params.name,
        keyPrefix: key.slice(0, 8),
        keyHash,
        permissions: params.permissions,
        expiresAt: params.expiresIn 
          ? new Date(Date.now() + params.expiresIn * 24 * 3600 * 1000)
          : null,
        status: 'active',
      },
    });
    
    this.logger.info('API Key created', {
      apiKeyId: apiKey.id,
      tenantId: params.tenantId,
    });
    
    return {
      id: apiKey.id,
      key,                    // 仅返回一次
      name: apiKey.name,
      permissions: apiKey.permissions,
      expiresAt: apiKey.expiresAt,
      createdAt: apiKey.createdAt,
    };
  }
  
  /**
   * 查找 API Key
   */
  async findByHash(keyHash: string): Promise<ApiKey | null> {
    // 1. 先查缓存
    const cached = await this.redis.get(`apikey:hash:${keyHash}`);
    if (cached) {
      return cached === 'null' ? null : JSON.parse(cached);
    }
    
    // 2. 查数据库
    const apiKey = await this.db.apiKey.findFirst({
      where: { keyHash, status: 'active' },
      include: { user: true },
    });
    
    // 3. 缓存结果
    await this.redis.setex(
      `apikey:hash:${keyHash}`,
      300,  // 5 分钟缓存
      JSON.stringify(apiKey)
    );
    
    return apiKey;
  }
  
  /**
   * 撤销 API Key
   */
  async revoke(apiKeyId: string): Promise<void> {
    const apiKey = await this.db.apiKey.update({
      where: { id: apiKeyId },
      data: { status: 'revoked' },
    });
    
    // 清除缓存
    await this.redis.del(`apikey:hash:${apiKey.keyHash}`);
    
    this.logger.info('API Key revoked', { apiKeyId });
  }
  
  /**
   * 记录使用
   */
  async recordUsage(apiKeyId: string): Promise<void> {
    await this.db.apiKey.update({
      where: { id: apiKeyId },
      data: {
        usageCount: { increment: 1 },
        lastUsedAt: new Date(),
      },
    });
  }
  
  /**
   * 生成 Key
   */
  private generateKey(): string {
    const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
    let key = this.KEY_PREFIX;
    
    for (let i = 0; i < this.KEY_LENGTH; i++) {
      key += chars.charAt(Math.floor(Math.random() * chars.length));
    }
    
    return key;
  }
  
  /**
   * 哈希 Key
   */
  private hashKey(key: string): string {
    // 使用单独的盐值
    const salt = process.env.API_KEY_SALT || 'default-salt';
    return crypto
      .createHash('sha256')
      .update(key + salt)
      .digest('hex');
  }
}
```
