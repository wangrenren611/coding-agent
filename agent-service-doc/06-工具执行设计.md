## 1. 模块概述

### 1.1 设计目标

工具执行模块是企业级 Agent 服务端的核心组件之一，负责：

| 目标 | 描述 | 优先级 |
|------|------|--------|
| **工具注册** | 动态注册和管理可用工具 | P0 |
| **安全执行** | 分级安全策略，防止恶意操作 | P0 |
| **资源控制** | 超时、内存、输出大小限制 | P0 |
| **可观测** | 完整的执行日志和指标 | P1 |
| **可扩展** | 支持自定义工具 | P1 |

### 1.2 核心组件

```
工具执行模块
│
├── Tool Registry (工具注册表)
│   ├── 工具定义管理
│   ├── 工具发现
│   └── 工具版本控制
│
├── Tool Executor (工具执行器)
│   ├── 执行调度
│   ├── 资源限制
│   └── 结果处理
│
├── Security Policy (安全策略)
│   ├── 安全级别定义
│   ├── 权限检查
│   └── 危险操作拦截
│
└── Sandbox Manager (沙箱管理)
    ├── 隔离环境池
    ├── 资源监控
    └── 清理回收
```

### 2.1 工具元数据

```typescript
// 工具定义
interface ToolDefinition {
  // 唯一标识
  name: string;
  
  // 描述
  description: string;
  category: ToolCategory;
  
  // 参数定义 (JSON Schema)
  parameters: JsonSchema;
  
  // 安全配置
  security: ToolSecurity;
  
  // 资源限制
  limits: ToolLimits;
  
  // 能力标记
  capabilities: {
    streaming?: boolean;       // 支持流式输出
    streamingPartial?: boolean; // 支持部分流式
    destructive?: boolean;    // 可能造成破坏
    idempotent?: boolean;     // 幂等操作
  };
  
  // 版本
  version: string;
  
  // 创建者 (自定义工具)
  createdBy?: string;
  
  // 时间戳
  createdAt: Date;
  updatedAt: Date;
}

// 工具分类
type ToolCategory = 
  | 'file'          // 文件操作
  | 'bash'          // 命令执行
  | 'search'        // 搜索
  | 'code'          // 代码相关
  | 'web'           // 网络请求
  | 'custom';       // 自定义

// 安全级别
enum SecurityLevel {
  SAFE = 'safe',           // 安全操作，无需特殊处理
  LIMITED = 'limited',     // 有限制，需要配置白名单
  SANDBOXED = 'sandboxed', // 沙箱执行，资源限制
  ISOLATED = 'isolated',   // 完全隔离 (Docker/VM)
}

// 资源限制
interface ToolLimits {
  // 超时限制
  timeout: number;           // 毫秒
  timeoutPerInput?: number;  // 每个输入的超时
  
  // 内存限制
  memory?: number;           // MB
  
  // 输出限制
  maxOutputSize?: number;   // 字节
  maxFileSize?: number;     // 字节
  
  // 文件系统限制
  allowedPaths?: string[];  // 允许访问的路径
  forbiddenPaths?: string[]; // 禁止访问的路径
  forbiddenPatterns?: string[]; // 禁止的文件模式
}
```

## 3. 工具注册表

### 3.1 核心接口

```typescript
// 工具注册表接口
interface IToolRegistry {
  // 注册工具
  register(tool: ToolDefinition): Promise<void>;
  
  // 注销工具
  unregister(name: string): Promise<void>;
  
  // 获取工具
  get(name: string): ToolDefinition | undefined;
  
  // 列出工具
  list(options?: ListToolsOptions): ToolDefinition[];
  
  // 检查工具是否存在
  has(name: string): boolean;
  
  // 获取工具 Schema (用于 LLM)
  getToolsForLLM(): LLMFunction[];
}

// 列出工具选项
interface ListToolsOptions {
  category?: ToolCategory;
  securityLevel?: SecurityLevel;
  search?: string;
}
```

### 3.2 实现

```typescript
/**
 * 工具注册表实现
 */
class ToolRegistry implements IToolRegistry {
  private tools = new Map<string, ToolDefinition>();
  private index = {
    byCategory: new Map<ToolCategory, Set<string>>(),
    bySecurityLevel: new Map<SecurityLevel, Set<string>>(),
  };
  
  constructor() {
    // 初始化内置工具
    this.initializeBuiltinTools();
  }
  
  /**
   * 注册工具
   */
  async register(tool: ToolDefinition): Promise<void> {
    // 验证工具定义
    this.validateTool(tool);
    
    // 检查是否已存在
    if (this.tools.has(tool.name)) {
      throw new ToolAlreadyExistsError(tool.name);
    }
    
    // 注册
    this.tools.set(tool.name, tool);
    
    // 更新索引
    this.addToIndex(tool);
    
    // 记录日志
    console.log(`[ToolRegistry] Registered tool: ${tool.name}`);
  }
  
  /**
   * 注销工具
   */
  async unregister(name: string): Promise<void> {
    const tool = this.tools.get(name);
    if (!tool) {
      throw new ToolNotFoundError(name);
    }
    
    // 禁止注销内置工具
    if (this.isBuiltinTool(name)) {
      throw new CannotUnregisterBuiltinToolError(name);
    }
    
    this.tools.delete(name);
    this.removeFromIndex(tool);
    
    console.log(`[ToolRegistry] Unregistered tool: ${name}`);
  }
  
  /**
   * 获取工具
   */
  get(name: string): ToolDefinition | undefined {
    return this.tools.get(name);
  }
  
  /**
   * 列出工具
   */
  list(options?: ListToolsOptions): ToolDefinition[] {
    let results = Array.from(this.tools.values());
    
    // 按分类过滤
    if (options?.category) {
      const allowed = this.index.byCategory.get(options.category);
      if (allowed) {
        results = results.filter(t => allowed.has(t.name));
      }
    }
    
    // 按安全级别过滤
    if (options?.securityLevel) {
      results = results.filter(t => t.security.level === options.securityLevel);
    }
    
    // 搜索过滤
    if (options?.search) {
      const searchLower = options.search.toLowerCase();
      results = results.filter(t => 
        t.name.toLowerCase().includes(searchLower) ||
        t.description.toLowerCase().includes(searchLower)
      );
    }
    
    return results;
  }
  
  /**
   * 检查工具是否存在
   */
  has(name: string): boolean {
    return this.tools.has(name);
  }
  
  /**
   * 获取工具 Schema (用于 LLM)
   */
  getToolsForLLM(): LLMFunction[] {
    return Array.from(this.tools.values()).map(tool => ({
      type: 'function',
      function: {
        name: tool.name,
        description: tool.description,
        parameters: tool.parameters,
      },
    }));
  }
  
  /**
   * 验证工具定义
   */
  private validateTool(tool: ToolDefinition): void {
    if (!tool.name || !/^[a-z][a-z0-9_]*$/.test(tool.name)) {
      throw new InvalidToolNameError(tool.name);
    }
    
    if (!tool.description) {
      throw new InvalidToolDefinitionError('description is required');
    }
    
    if (!tool.parameters || !tool.parameters.type) {
      throw new InvalidToolDefinitionError('parameters must be a valid JSON schema');
    }
    
    // 验证安全配置
    if (!tool.security?.level) {
      throw new InvalidToolDefinitionError('security level is required');
    }
  }
  
  /**
   * 添加到索引
   */
  private addToIndex(tool: ToolDefinition): void {
    // 按分类索引
    if (!this.index.byCategory.has(tool.category)) {
      this.index.byCategory.set(tool.category, new Set());
    }
    this.index.byCategory.get(tool.category)!.add(tool.name);
    
    // 按安全级别索引
    if (!this.index.bySecurityLevel.has(tool.security.level)) {
      this.index.bySecurityLevel.set(tool.security.level, new Set());
    }
    this.index.bySecurityLevel.get(tool.security.level)!.add(tool.name);
  }
  
  /**
   * 从索引移除
   */
  private removeFromIndex(tool: ToolDefinition): void {
    this.index.byCategory.get(tool.category)?.delete(tool.name);
    this.index.bySecurityLevel.get(tool.security.level)?.delete(tool.name);
  }
  
  /**
   * 初始化内置工具
   */
  private initializeBuiltinTools(): void {
    for (const tool of BUILTIN_TOOLS) {
      this.tools.set(tool.name, tool);
      this.addToIndex(tool);
    }
  }
  
  /**
   * 判断是否为内置工具
   */
  private isBuiltinTool(name: string): boolean {
    return BUILTIN_TOOLS.some(t => t.name === name);
  }
}
```

### 4.2 核心接口

```typescript
// 工具执行器接口
interface IToolExecutor {
  // 执行工具
  execute(
    sessionId: string,
    toolName: string,
    args: Record<string, any>,
    context: ExecutionContext
  ): Promise<ToolExecutionResult>;
  
  // 流式执行
  executeStream(
    sessionId: string,
    toolName: string,
    args: Record<string, any>,
    context: ExecutionContext,
    onChunk: (chunk: ToolStreamChunk) => void
  ): Promise<ToolExecutionResult>;
  
  // 中止执行
  abort(executionId: string): Promise<void>;
}

// 执行上下文
interface ExecutionContext {
  sessionId: string;
  userId: string;
  tenantId: string;
  workspace?: string;       // 工作目录
  allowedPaths?: string[]; // 允许访问的路径
  signal?: AbortSignal;   // 中止信号
}

// 执行结果
interface ToolExecutionResult {
  executionId: string;
  toolName: string;
  success: boolean;
  result?: any;
  error?: ToolExecutionError;
  metadata: {
    duration: number;       // 毫秒
    memoryUsed?: number;   // MB
    outputSize?: number;   // 字节
  };
}

// 流式块
interface ToolStreamChunk {
  type: 'stdout' | 'stderr' | 'error' | 'done';
  data: string;
  timestamp: number;
}
```

### 4.3 实现

```typescript
/**
 * 工具执行器实现
 */
class ToolExecutor implements IToolExecutor {
  private registry: ToolRegistry;
  private sandboxPool: SandboxPool;
  private executors = new Map<SecurityLevel, BaseExecutor>();
  private executions = new Map<string, AbortController>();
  private metrics: MetricsCollector;
  
  constructor(
    registry: ToolRegistry,
    sandboxPool: SandboxPool,
    metrics: MetricsCollector
  ) {
    this.registry = registry;
    this.sandboxPool = sandboxPool;
    this.metrics = metrics;
    
    // 初始化执行器
    this.initializeExecutors();
  }
  
  /**
   * 执行工具
   */
  async execute(
    sessionId: string,
    toolName: string,
    args: Record<string, any>,
    context: ExecutionContext
  ): Promise<ToolExecutionResult> {
    const executionId = generateUUID();
    const startTime = Date.now();
    
    try {
      // 1. 获取工具定义
      const tool = this.registry.get(toolName);
      if (!tool) {
        throw new ToolNotFoundError(toolName);
      }
      
      // 2. 安全检查
      this.checkSecurity(tool, args, context);
      
      // 3. 参数验证
      this.validateParameters(tool, args);
      
      // 4. 创建中止控制器
      const controller = new AbortController();
      this.executions.set(executionId, controller);
      
      // 5. 设置超时
      const timeout = args.timeout ?? tool.limits.timeout;
      const timeoutId = setTimeout(() => controller.abort(), timeout);
      
      // 6. 选择执行器并执行
      const executor = this.getExecutor(tool.security.level);
      const result = await executor.execute({
        tool,
        args,
        context: { ...context, signal: controller.signal },
      });
      
      clearTimeout(timeoutId);
      
      // 7. 记录指标
      this.metrics.recordToolExecution({
        toolName,
        success: true,
        duration: Date.now() - startTime,
        sessionId,
      });
      
      return {
        executionId,
        toolName,
        success: true,
        result,
        metadata: {
          duration: Date.now() - startTime,
        },
      };
      
    } catch (error) {
      // 记录错误指标
      this.metrics.recordToolExecution({
        toolName,
        success: false,
        duration: Date.now() - startTime,
        sessionId,
        error: error.message,
      });
      
      return {
        executionId,
        toolName,
        success: false,
        error: this.normalizeError(error),
        metadata: {
          duration: Date.now() - startTime,
        },
      };
      
    } finally {
      this.executions.delete(executionId);
    }
  }
  
  /**
   * 流式执行
   */
  async executeStream(
    sessionId: string,
    toolName: string,
    args: Record<string, any>,
    context: ExecutionContext,
    onChunk: (chunk: ToolStreamChunk) => void
  ): Promise<ToolExecutionResult> {
    const tool = this.registry.get(toolName);
    if (!tool) {
      throw new ToolNotFoundError(toolName);
    }
    
    if (!tool.capabilities.streaming) {
      throw new ToolDoesNotSupportStreamingError(toolName);
    }
    
    const executor = this.getExecutor(tool.security.level);
    if (!('executeStream' in executor)) {
      throw new ToolDoesNotSupportStreamingError(toolName);
    }
    
    return executor.executeStream({ tool, args, context }, onChunk);
  }
  
  /**
   * 中止执行
   */
  async abort(executionId: string): Promise<void> {
    const controller = this.executions.get(executionId);
    if (controller) {
      controller.abort();
    }
  }
  
  /**
   * 安全检查
   */
  private checkSecurity(
    tool: ToolDefinition,
    args: Record<string, any>,
    context: ExecutionContext
  ): void {
    const { security } = tool;
    
    // 1. 检查路径限制
    if (security.allowedPaths || security.forbiddenPaths) {
      this.checkPathSecurity(args, security, context);
    }
    
    // 2. 检查命令限制 (bash)
    if (security.forbiddenCommands) {
      this.checkCommandSecurity(args, security.forbiddenCommands);
    }
    
    // 3. 检查 URL 限制 (web)
    if (security.forbiddenUrls) {
      this.checkUrlSecurity(args, security.forbiddenUrls);
    }
  }
  
  /**
   * 检查路径安全
   */
  private checkPathSecurity(
    args: Record<string, any>,
    security: ToolSecurity,
    context: ExecutionContext
  ): void {
    const filePaths = this.extractFilePaths(args);
    
    for (const filePath of filePaths) {
      const resolvedPath = this.resolvePath(filePath, context);
      
      // 检查禁止路径
      if (security.forbiddenPaths) {
        for (const forbidden of security.forbiddenPaths) {
          if (this.isSubPath(resolvedPath, forbidden)) {
            throw new PathAccessDeniedError(filePath, forbidden);
          }
        }
      }
      
      // 检查允许路径
      if (security.allowedPaths) {
        let isAllowed = false;
        for (const allowed of security.allowedPaths) {
          const resolvedAllowed = this.resolvePath(allowed, context);
          if (this.isSubPath(resolvedPath, resolvedAllowed)) {
            isAllowed = true;
            break;
          }
        }
        if (!isAllowed) {
          throw new PathNotAllowedError(filePath, security.allowedPaths);
        }
      }
    }
  }
  
  /**
   * 检查命令安全
   */
  private checkCommandSecurity(
    args: Record<string, any>,
    forbiddenCommands: string[]
  ): void {
    const command = args.command;
    if (!command) return;
    
    for (const forbidden of forbiddenCommands) {
      if (this.matchPattern(command, forbidden)) {
        throw new CommandNotAllowedError(forbidden);
      }
    }
  }
  
  /**
   * 选择执行器
   */
  private getExecutor(level: SecurityLevel): BaseExecutor {
    const executor = this.executors.get(level);
    if (!executor) {
      throw new UnsupportedSecurityLevelError(level);
    }
    return executor;
  }
  
  /**
   * 初始化执行器
   */
  private initializeExecutors(): void {
    this.executors.set(SecurityLevel.SAFE, new SafeExecutor(this.registry));
    this.executors.set(SecurityLevel.LIMITED, new LimitedExecutor(this.registry));
    this.executors.set(SecurityLevel.SANDBOXED, new SandboxedExecutor(this.registry, this.sandboxPool));
    this.executors.set(SecurityLevel.ISOLATED, new IsolatedExecutor(this.sandboxPool));
  }
}
```

```typescript
/**
 * 安全执行器 - 用于无任何风险的操作为
 */
class SafeExecutor extends BaseExecutor {
  async execute(params: ExecuteParams): Promise<any> {
    const { tool, args } = params;
    
    switch (tool.name) {
      case 'web_search':
        return this.executeWebSearch(args);
        
      case 'web_fetch':
        return this.executeWebFetch(args);
        
      case 'glob':
        return this.executeGlob(params);
        
      default:
        throw new ToolExecutionError(`Unknown safe tool: ${tool.name}`);
    }
  }
  
  private async executeWebSearch(args: { query: string; maxResults?: number }): Promise<any> {
    const Tavily = require('@tavily/core');
    const tavily = new Tavily({ apiKey: process.env.TAVILY_API_KEY });
    
    const results = await tavily.search(args.query, {
      maxResults: args.maxResults ?? 5,
    });
    
    return {
      results: results.results.map((r: any) => ({
        title: r.title,
        url: r.url,
        content: r.content,
        score: r.score,
      })),
    };
  }
  
  private async executeWebFetch(args: { url: string; format?: string }): Promise<any> {
    const response = await fetch(args.url, {
      headers: {
        'User-Agent': 'Agent-Service/1.0',
      },
    });
    
    const contentType = response.headers.get('content-type') || '';
    
    if (args.format === 'html' || contentType.includes('text/html')) {
      const TurndownService = require('turndown');
      const turndown = new TurndownService();
      const html = await response.text();
      return { content: turndown.turndown(html) };
    }
    
    return { content: await response.text() };
  }
  
  private async executeGlob(params: ExecuteParams): Promise<any> {
    const { args, context } = params;
    const fg = require('fast-glob');
    
    const pattern = args.pattern;
    const basePath = args.basePath || context.workspace || '.';
    
    const files = await fg(pattern, {
      cwd: basePath,
      absolute: true,
    });
    
    return { files };
  }
}
```

### 5.2 受限执行器 (Limited Executor)

```typescript
/**
 * 受限执行器 - 用于需要路径限制的文件操作
 */
class LimitedExecutor extends SafeExecutor {
  async execute(params: ExecuteParams): Promise<any> {
    const { tool, args, context } = params;
    
    switch (tool.name) {
      case 'read_file':
        return this.executeReadFile(args, context);
        
      case 'write_file':
        return this.executeWriteFile(args, context);
        
      case 'grep':
        return this.executeGrep(params);
        
      default:
        return super.execute(params);
    }
  }
  
  private async executeReadFile(
    args: { filePath: string; offset?: number; limit?: number },
    context: ExecutionContext
  ): Promise<any> {
    const fs = require('fs/promises');
    const path = require('path');
    
    // 解析路径
    const filePath = this.resolvePath(args.filePath, context);
    
    // 检查是否存在
    const stats = await fs.stat(filePath);
    
    // 检查文件大小
    if (args.limit && stats.size > args.limit) {
      // 分块读取
      const fd = await fs.open(filePath, 'r');
      const buffer = Buffer.alloc(args.limit);
      await fd.read(buffer, 0, args.limit, args.offset || 0);
      await fd.close();
      
      return {
        content: buffer.toString('utf-8'),
        truncated: true,
        totalSize: stats.size,
        readSize: args.limit,
      };
    }
    
    // 读取完整内容
    const content = await fs.readFile(filePath, 'utf-8');
    
    return {
      content,
      totalSize: stats.size,
    };
  }
  
  private async executeWriteFile(
    args: { filePath: string; content: string; mode?: 'overwrite' | 'append' },
    context: ExecutionContext
  ): Promise<any> {
    const fs = require('fs/promises');
    const path = require('path');
    
    const filePath = this.resolvePath(args.filePath, context);
    
    // 确保目录存在
    await fs.mkdir(path.dirname(filePath), { recursive: true });
    
    // 写入文件
    const mode = args.mode || 'overwrite';
    if (mode === 'append') {
      await fs.appendFile(filePath, args.content, 'utf-8');
    } else {
      await fs.writeFile(filePath, args.content, 'utf-8');
    }
    
    return {
      success: true,
      path: filePath,
    };
  }
}
```

```typescript
/**
 * 沙箱执行器 - 用于危险操作，使用 Node.js VM 隔离
 */
class SandboxedExecutor extends BaseExecutor {
  async execute(params: ExecuteParams): Promise<any> {
    const { tool, args, context } = params;
    
    switch (tool.name) {
      case 'bash':
        return this.executeBash(args, context);
        
      default:
        throw new ToolExecutionError(`Sandboxed tool not supported: ${tool.name}`);
    }
  }
  
  private async executeBash(
    args: { command: string; timeout?: number; workingDirectory?: string },
    context: ExecutionContext
  ): Promise<any> {
    const { execa } = require('execa');
    
    // 构建命令
    const command = args.command;
    const cwd = args.workingDirectory || context.workspace || process.cwd();
    
    try {
      const result = await execa(command, {
        cwd,
        shell: true,
        timeout: args.timeout || 60000,
        maxBuffer: 1024 * 1024,  // 1MB
        stripFinalNewline: true,
      });
      
      return {
        stdout: result.stdout,
        stderr: result.stderr,
        exitCode: result.exitCode,
        failed: result.failed,
      };
      
    } catch (error: any) {
      // 超时
      if (error.timedOut) {
        throw new ToolExecutionError('Command timed out', { timeout: true });
      }
      
      // 返回错误结果
      return {
        stdout: error.stdout || '',
        stderr: error.stderr || error.message,
        exitCode: error.exitCode || 1,
        failed: true,
        timedOut: error.timedOut,
      };
    }
  }
}
```

### 6.1 沙箱池管理

```typescript
/**
 * 沙箱池管理器 - 管理隔离执行环境
 */
class SandboxPool {
  private pools = new Map<SecurityLevel, Sandbox[]>();
  private metrics: MetricsCollector;
  
  constructor(metrics: MetricsCollector) {
    this.metrics = metrics;
    this.initializePools();
  }
  
  /**
   * 初始化沙箱池
   */
  private initializePools(): void {
    // 预创建沙箱实例
    this.pools.set(SecurityLevel.SANDBOXED, []);
    this.pools.set(SecurityLevel.ISOLATED, []);
  }
  
  /**
   * 获取沙箱实例
   */
  async acquire(level: SecurityLevel): Promise<Sandbox> {
    const pool = this.pools.get(level);
    if (!pool) {
      throw new UnsupportedSecurityLevelError(level);
    }
    
    // 尝试从池中获取可用沙箱
    const sandbox = pool.pop();
    if (sandbox && sandbox.isHealthy()) {
      return sandbox;
    }
    
    // 创建新的沙箱实例
    return this.createSandbox(level);
  }
  
  /**
   * 归还沙箱实例
   */
  async release(sandbox: Sandbox): Promise<void> {
    const pool = this.pools.get(sandbox.level);
    if (!pool) {
      await sandbox.destroy();
      return;
    }
    
    // 重置沙箱状态
    await sandbox.reset();
    
    // 如果池未满，归还池中
    if (pool.length < this.getPoolSize(sandbox.level)) {
      pool.push(sandbox);
    } else {
      await sandbox.destroy();
    }
  }
  
  /**
   * 创建沙箱实例
   */
  private async createSandbox(level: SecurityLevel): Promise<Sandbox> {
    switch (level) {
      case SecurityLevel.SANDBOXED:
        return new NodeSandbox();
      case SecurityLevel.ISOLATED:
        return new DockerSandbox();
      default:
        throw new UnsupportedSecurityLevelError(level);
    }
  }
  
  /**
   * 获取池大小
   */
  private getPoolSize(level: SecurityLevel): number {
    // 根据 CPU 核心数动态计算
    const cpuCount = os.cpus().length;
    switch (level) {
      case SecurityLevel.SANDBOXED:
        return cpuCount * 2;
      case SecurityLevel.ISOLATED:
        return Math.max(2, cpuCount);
      default:
        return 0;
    }
  }
}

interface Sandbox {

class NodeSandbox implements Sandbox {

  private vm: vm.VM;
  
  constructor() {
    this.vm = new vm.VM({
      timeout: 30000,
      sandbox: {},
    });
  }

    return true;
  }

    // 使用 child_process 执行命令
    const { execa } = require('execa');
    const result = await execa('bash', ['-c', command], {
      cwd: options.workingDirectory,
      timeout: options.timeout,
    });
    return result;
  }
  
  async reset(): Promise<void> {
    // 清理临时文件等
  }
  
  async destroy(): Promise<void> {
    // 销毁沙箱
  }
}

/**
 * Docker 沙箱 - 用于完全隔离的执行
 */
class DockerSandbox implements Sandbox {
  readonly level = SecurityLevel.ISOLATED;
  private containerId?: string;
  
  async execute(command: string, options: ExecuteOptions): Promise<ExecuteResult> {
    // 创建临时容器执行命令
    const container = await docker.createContainer({
      Image: 'agent-sandbox:latest',
      Cmd: ['sh', '-c', command],
      WorkingDir: options.workingDirectory,
      NetworkMode: 'none',
      Memory: 512 * 1024 * 1024, // 512MB
      CpuShares: 512,
    });
    
    this.containerId = container.id;
    
    try {
      await container.start();
      const result = await container.wait();
      const logs = await container.logs({ stdout: true, stderr: true });
      
      return {
        stdout: logs.stdout,
        stderr: logs.stderr,
        exitCode: result.StatusCode,
      };
    } finally {
      await container.remove({ force: true });
    }
  }
  
  isHealthy(): boolean {
    return true;
  }
  
  async reset(): Promise<void> {
    // Docker 容器天然隔离，无需重置
  }
  
  async destroy(): Promise<void> {
    if (this.containerId) {
      try {
        const container = docker.getContainer(this.containerId);
        await container.remove({ force: true });
      } catch (e) {
        // 忽略错误
      }
    }
  }
}
```

```typescript
// 路径检查函数
function checkPathAccess(
  requestedPath: string,
  allowedPaths: string[],
  forbiddenPaths: string[]
): void {
  const resolved = path.resolve(requestedPath);
  
  // 检查禁止路径
  for (const forbidden of forbiddenPaths) {
    if (isSubPath(resolved, forbidden)) {
      throw new PathAccessDeniedError(resolved, forbidden);
    }
  }
  
  // 检查允许路径
  if (allowedPaths.length > 0) {
    let isAllowed = false;
    for (const allowed of allowedPaths) {
      if (isSubPath(resolved, path.resolve(allowed))) {
        isAllowed = true;
        break;
      }
    }
    if (!isAllowed) {
      throw new PathNotAllowedError(resolved, allowedPaths);
    }
  }
}

function isSubPath(child: string, parent: string): boolean {
  const relative = path.relative(parent, child);
  return !relative.startsWith('..') && !path.isAbsolute(relative);
}
```
