## 1. 数据库概述

### 1.1 技术选型

| 组件 | 技术 | 版本 | 用途 |
|------|------|------|------|
| **主数据库** | PostgreSQL | ≥15 | 核心数据存储 |
| **ORM** | Prisma | ≥5.0 | 数据访问层 |
| **缓存** | Redis | ≥7.0 | 热点数据缓存 |

### 1.2 数据库架构

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                           数据库架构                                        │
└─────────────────────────────────────────────────────────────────────────────┘

┌──────────────┐     ┌──────────────┐     ┌──────────────┐
│   租户/用户   │     │   会话/消息   │     │   系统数据    │
│    (小表)    │     │    (大表)    │     │    (小表)    │
└──────┬───────┘     └──────┬───────┘     └──────┬───────┘
       │                    │                    │
       │         ┌───────────┼───────────┐        │
       │         ▼           ▼           ▼        │
       │    ┌────────┐  ┌────────┐  ┌────────┐   │
       │    │ Sessions│  │Messages│  │Compacts│   │
       │    └────────┘  └────────┘  └────────┘   │
       │         │           │           │        │
       └─────────┴───────────┴───────────┴────────┘
                         │
                         ▼
              ┌────────────────────┐
              │   PostgreSQL       │
              │  (分区表策略)      │
              └────────────────────┘

┌─────────────────────────────────────────────────────────────────────────────┐
│                              缓存层 (Redis)                                 │
├─────────────────────────────────────────────────────────────────────────────┤
│  会话缓存 │ 上下文缓存 │ 消息队列 │ 限流计数 │ Token 黑名单 │ 分布式锁      │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 1.3 命名规范

| 规则 | 示例 |
|------|------|
| 表名 | snake_case, 复数形式 `sessions`, `users` |
| 列名 | snake_case `created_at`, `user_id` |
| 主键 | `id` (UUID) |
| 外键 | `{table}_id` `session_id`, `user_id` |
| 索引 | `idx_{table}_{columns}` |
| 唯一约束 | `uk_{table}_{columns}` |

## 2. 表结构详细设计

### 2.1 租户表 (tenants)

```sql
-- 租户表
CREATE TABLE tenants (
    -- 主键
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    
    -- 基本信息
    name VARCHAR(255) NOT NULL,
    slug VARCHAR(100),  -- URL 友好标识
    email VARCHAR(255) NOT NULL,
    
    -- 套餐与配额
    plan VARCHAR(50) NOT NULL DEFAULT 'free' 
        CHECK (plan IN ('free', 'starter', 'pro', 'enterprise')),
    
    -- 自定义配额 (JSON)
    custom_quotas JSONB DEFAULT '{}',
    
    -- 状态
    status VARCHAR(50) NOT NULL DEFAULT 'active'
        CHECK (status IN ('pending', 'active', 'suspended', 'deleted')),
    
    -- 功能开关
    features JSONB DEFAULT '{"customTools": false, "apiAccess": true}',
    
    -- 设置
    settings JSONB DEFAULT '{}',
    
    -- 时间戳
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 唯一约束
CREATE UNIQUE INDEX uk_tenants_slug ON tenants(slug) WHERE slug IS NOT NULL;
CREATE UNIQUE INDEX uk_tenants_email ON tenants(email);

-- 索引
CREATE INDEX idx_tenants_status ON tenants(status);
CREATE INDEX idx_tenants_plan ON tenants(plan);
```

**Prisma Schema:**

```prisma
model Tenant {
    id            String    @id @default(uuid())
    name          String
    slug          String?   @unique
    email         String    @unique
    plan          Plan      @default(FREE)
    customQuotas  Json?     @default("{}")
    status        TenantStatus @default(ACTIVE)
    features      Json?     @default("{}")
    settings      Json?     @default("{}")
    
    createdAt     DateTime  @default(now())
    updatedAt     DateTime  @updatedAt
    
    // 关系
    users         User[]
    sessions      Session[]
    apiKeys       ApiKey[]
    usageRecords  UsageRecord[]
    
    @@index([status])
    @@index([plan])
}

enum Plan {
    FREE
    STARTER
    PRO
    ENTERPRISE
}

enum TenantStatus {
    PENDING
    ACTIVE
    SUSPENDED
    DELETED
}
```

### 2.2 用户表 (users)

```sql
-- 用户表
CREATE TABLE users (
    -- 主键
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    
    -- 租户关联
    tenant_id UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
    
    -- 基本信息
    email VARCHAR(255) NOT NULL,
    name VARCHAR(255),
    avatar VARCHAR(500),
    
    -- 认证
    password_hash VARCHAR(255),
    
    -- 角色
    role VARCHAR(50) NOT NULL DEFAULT 'user'
        CHECK (role IN ('owner', 'admin', 'developer', 'user')),
    
    -- 状态
    status VARCHAR(50) NOT NULL DEFAULT 'active'
        CHECK (status IN ('active', 'invited', 'disabled')),
    
    -- 时间戳
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    last_login_at TIMESTAMP WITH TIME ZONE,
    
    -- 约束
    CONSTRAINT uk_users_tenant_email UNIQUE (tenant_id, email)
);

-- 索引
CREATE INDEX idx_users_tenant ON users(tenant_id);
CREATE INDEX idx_users_status ON users(status);
CREATE INDEX idx_users_email ON users(email);
```

**Prisma Schema:**

```prisma
model User {
    id            String    @id @default(uuid())
    tenantId      String
    email         String
    name          String?
    avatar        String?
    passwordHash  String?
    role          UserRole  @default(USER)
    status        UserStatus @default(ACTIVE)
    
    createdAt     DateTime  @default(now())
    updatedAt     DateTime  @updatedAt
    lastLoginAt   DateTime?
    
    // 关系
    tenant        Tenant    @relation(fields: [tenantId], references: [id], onDelete: Cascade)
    sessions      Session[]
    apiKeys       ApiKey[]
    
    @@unique([tenantId, email])
    @@index([tenantId])
    @@index([status])
}

enum UserRole {
    OWNER
    ADMIN
    DEVELOPER
    USER
}

enum UserStatus {
    ACTIVE
    INVITED
    DISABLED
}
```

### 2.3 会话表 (sessions)

```sql
-- 会话表
CREATE TABLE sessions (
    -- 主键
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    
    -- 租户/用户关联
    tenant_id UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    
    -- 配置
    system_prompt TEXT NOT NULL,
    model VARCHAR(100) DEFAULT 'gpt-4',
    tools JSONB DEFAULT '[]',
    temperature DECIMAL(3,2),
    max_tokens INTEGER,
    metadata JSONB DEFAULT '{}',
    
    -- 状态
    status VARCHAR(50) NOT NULL DEFAULT 'active'
        CHECK (status IN ('active', 'idle', 'archived', 'deleted')),
    
    -- 计量
    prompt_tokens BIGINT DEFAULT 0,
    completion_tokens BIGINT DEFAULT 0,
    total_tokens BIGINT DEFAULT 0,
    request_count INTEGER DEFAULT 0,
    message_count INTEGER DEFAULT 0,
    
    -- 时间戳
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    last_active_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    expires_at TIMESTAMP WITH TIME ZONE,
    
    -- 约束
    CONSTRAINT uk_sessions_tenant_user UNIQUE (tenant_id, id)
);

-- 索引
CREATE INDEX idx_sessions_tenant ON sessions(tenant_id);
CREATE INDEX idx_sessions_user ON sessions(user_id);
CREATE INDEX idx_sessions_status ON sessions(status);
CREATE INDEX idx_sessions_created ON sessions(created_at DESC);
CREATE INDEX idx_sessions_last_active ON sessions(last_active_at DESC);
CREATE INDEX idx_sessions_expires ON sessions(expires_at) WHERE expires_at IS NOT NULL;
```

**Prisma Schema:**

```prisma
model Session {
    id            String    @id @default(uuid())
    tenantId      String
    userId        String
    
    // 配置
    systemPrompt  String
    model         String    @default("gpt-4")
    tools         Json?     @default("[]")
    temperature   Float?
    maxTokens     Int?
    metadata      Json?     @default("{}")
    
    // 状态
    status        SessionStatus @default(ACTIVE)
    
    // 计量
    promptTokens      BigInt   @default(0)
    completionTokens  BigInt   @default(0)
    totalTokens       BigInt   @default(0)
    requestCount      Int      @default(0)
    messageCount      Int      @default(0)
    
    // 时间戳
    createdAt     DateTime  @default(now())
    updatedAt     DateTime  @updatedAt
    lastActiveAt  DateTime  @default(now())
    expiresAt     DateTime?
    
    // 关系
    tenant        Tenant    @relation(fields: [tenantId], references: [id], onDelete: Cascade)
    user          User      @relation(fields: [userId], references: [id], onDelete: Cascade)
    messages      Message[]
    context       Context?
    compactionRecords CompactionRecord[]
    
    @@index([tenantId])
    @@index([userId])
    @@index([status])
    @@index([createdAt(sort: Desc)])
    @@index([lastActiveAt])
}

enum SessionStatus {
    ACTIVE
    IDLE
    ARCHIVED
    DELETED
}
```

### 2.4 消息表 (messages)

```sql
-- 消息表 (分区表)
CREATE TABLE messages (
    -- 主键
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    
    -- 会话关联
    session_id UUID NOT NULL REFERENCES sessions(id) ON DELETE CASCADE,
    
    -- 角色
    role VARCHAR(20) NOT NULL 
        CHECK (role IN ('system', 'user', 'assistant', 'tool')),
    
    -- 内容
    content JSONB NOT NULL,
    content_text TEXT,  -- 纯文本，用于全文搜索
    
    -- 消息类型
    type VARCHAR(50),
    
    -- 工具调用
    tool_calls JSONB,
    tool_call_id UUID,
    
    -- 元数据
    metadata JSONB DEFAULT '{}',
    usage JSONB,
    
    -- 时间戳
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    
    -- 约束
    CONSTRAINT fk_messages_session FOREIGN KEY (session_id) 
        REFERENCES sessions(id) ON DELETE CASCADE
) PARTITION BY RANGE (created_at);

-- 创建分区 (按月)
CREATE TABLE messages_2026_01 PARTITION OF messages
    FOR VALUES FROM ('2026-01-01') TO ('2026-02-01');

CREATE TABLE messages_2026_02 PARTITION OF messages
    FOR VALUES FROM ('2026-02-01') TO ('2026-03-01');

CREATE TABLE messages_2026_03 PARTITION OF messages
    FOR VALUES FROM ('2026-03-01') TO ('2026-04-01');

-- 索引
CREATE INDEX idx_messages_session ON messages(session_id);
CREATE INDEX idx_messages_session_created ON messages(session_id, created_at DESC);
CREATE INDEX idx_messages_role ON messages(role);
CREATE INDEX idx_messages_content_text ON messages USING gin(content_text gin_trgm_ops);
CREATE INDEX idx_messages_tool_call_id ON messages(tool_call_id) WHERE tool_call_id IS NOT NULL;
```

**Prisma Schema:**

```prisma
model Message {
    id            String    @id @default(uuid())
    sessionId     String
    
    role          MessageRole
    content       Json
    contentText   String?   // 全文搜索
    
    type          String?
    
    toolCalls     Json?
    toolCallId    String?
    
    metadata      Json?     @default("{}")
    usage         Json?
    
    createdAt     DateTime  @default(now())
    
    // 关系
    session       Session   @relation(fields: [sessionId], references: [id], onDelete: Cascade)
    
    @@index([sessionId])
    @@index([sessionId, createdAt(sort: Desc)])
    @@index([role])
    @@index([toolCallId])
}

enum MessageRole {
    SYSTEM
    USER
    ASSISTANT
    TOOL
}
```

### 2.5 上下文表 (contexts)

```sql
-- 上下文表
CREATE TABLE contexts (
    -- 主键
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    
    -- 会话关联
    session_id UUID NOT NULL REFERENCES sessions(id) ON DELETE CASCADE UNIQUE,
    
    -- 版本控制
    version INTEGER NOT NULL DEFAULT 1,
    
    -- 消息 (简化存储，引用 messages 表)
    message_count INTEGER DEFAULT 0,
    
    -- 统计
    stats JSONB DEFAULT '{"totalTokens": 0, "lastCompactionAt": null}',
    
    -- 时间戳
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    
    -- 约束
    CONSTRAINT fk_contexts_session FOREIGN KEY (session_id) 
        REFERENCES sessions(id) ON DELETE CASCADE
);

-- 索引
CREATE INDEX idx_contexts_session ON contexts(session_id);
CREATE INDEX idx_contexts_version ON contexts(version);
```

**Prisma Schema:**

```prisma
model Context {
    id            String    @id @default(uuid())
    sessionId     String    @unique
    
    version       Int       @default(1)
    messageCount  Int       @default(0)
    stats         Json?     @default("{}")
    
    createdAt     DateTime  @default(now())
    updatedAt     DateTime  @updatedAt
    
    // 关系
    session       Session   @relation(fields: [sessionId], references: [id], onDelete: Cascade)
    
    @@index([sessionId])
    @@index([version])
}
```

### 2.7 API Key 表 (api_keys)

```sql
-- API Key 表
CREATE TABLE api_keys (

    -- 状态
    status VARCHAR(50) NOT NULL DEFAULT 'active'
        CHECK (status IN ('active', 'revoked')),
    
    -- 时间戳
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 索引
CREATE INDEX idx_api_keys_tenant ON api_keys(tenant_id);
```

## 3. Redis 数据结构

### 3.1 缓存键设计

```typescript
// 缓存键命名规范
const CACHE_KEYS = {
  // 会话缓存
  SESSION: 'session:{id}',                    // Session 对象
  SESSION_LIST: 'session:list:{tenantId}',     // 会话列表
  
  // 上下文缓存
  CONTEXT: 'context:{sessionId}',              // 当前上下文
  CONTEXT_VERSION: 'context:version:{sessionId}', // 版本号
  
  // 消息队列
  QUEUE: 'queue:session:{sessionId}',          // 消息队列
  QUEUE_DELAYED: 'queue:delayed:{sessionId}',  // 延迟队列
  
  // 限流
  RATE_LIMIT: 'ratelimit:{tenantId}:{userId}:{type}',  // 限流计数
  QUOTA: 'quota:{tenantId}:{type}:{date}',     // 配额使用
  CONCURRENT: 'ratelimit:concurrent:{tenantId}', // 并发计数
  
  // Token
  JWT_BLACKLIST: 'jwt:blacklist:{jti}',         // JWT 黑名单
  JWT_REFRESH: 'jwt:refresh:{jti}',            // Refresh Token
  
  // API Key
  API_KEY: 'apikey:{id}',                      // API Key 对象
  API_KEY_HASH: 'apikey:hash:{hash}',          // API Key 哈希
  
  // 锁
  LOCK: 'lock:{resource}',                     // 分布式锁
};
```

### 3.2 数据结构选择

| 数据类型 | 用途 | Redis 命令 |
|----------|------|------------|
| **String** | 简单值缓存 | GET/SET |
| **Hash** | 对象缓存 | HGET/HSET |
| **List** | 消息队列 | LPUSH/RPOP |
| **Sorted Set** | 限流滑动窗口 | ZADD/ZRANGE |
| **Set** | 标签、去重 | SADD/SMEMBERS |
| **Expire** | TTL 过期 | EXPIRE |

### 5.2 分区策略

```sql
-- 消息表按月分区
-- 优点：
-- 1. 高效删除历史数据 (DROP PARTITION)
-- 2. 减少索引大小
-- 3. 提高查询性能

-- 审计日志按月分区
-- 优点：
-- 1. 大表拆分
-- 2. 归档/删除方便

-- 创建自动分区管理函数
CREATE OR REPLACE FUNCTION create_message_partition()
RETURNS void AS $$
DECLARE
    partition_date TEXT;
    partition_name TEXT;
BEGIN
    -- 创建下个月的分区
    partition_date := to_char(CURRENT_DATE + INTERVAL '1 month', 'YYYY_MM');
    partition_name := 'messages_' || partition_date;
    
    EXECUTE format(
        'CREATE TABLE IF NOT EXISTS %I PARTITION OF messages FOR VALUES FROM (%L) TO (%L)',
        partition_name,
        (CURRENT_DATE + INTERVAL '1 month')::date,
        (CURRENT_DATE + INTERVAL '2 month')::date
    );
END;
$$ LANGUAGE plpgsql;
```
